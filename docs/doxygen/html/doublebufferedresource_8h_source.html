67  /// @brief A thread specific resource that uses double buffering to avoid multithreaded synchronization mechanisms.
68  /// @details It is intended for a Mezzanine::Threading::iWorkUnit "iWorkUnit" like the to provide asynchronous
69  /// access to resources that to not need immediate atomicity. For example, it is ok if logs are gathered asynchronously
70  /// and aggregated later as long no log messages are damaged, lost or taken out of order. This allows IO and other potentially
71  /// expensive tasks to be used without blocking the current thread.
72  template<typename T>
73  class DoubleBufferedResource
74  {
75  private:
76  /// @brief One of the resources to be buffered
77  T* ResourceA;
78 
79  /// @brief The other of the resources to be buffered
80  T* ResourceB;
81 
82  public:
83  /// @brief A Constructor that creates default versions of the resources.
84  /// @details This calls the default constructors for type T.
85  DoubleBufferedResource() : ResourceA(0), ResourceB(0)
86  {
87  try
88  {
89  ResourceA = new T;
90  ResourceB = new T;
91  }catch(std::exception& e){
92  if (ResourceA)
93  { delete ResourceA; ResourceA=0; }
94  if (ResourceB)
95  { delete ResourceB; ResourceB=0; }
96  throw e;
97  }
98 
99  }
100 
101  /// @brief A constructor that takes ownership of the the resources passed.
102  /// @param Current This will be used as the first resource for threads to use.
103  /// @param Buffer This will wait for the other thread
104  DoubleBufferedResource(T* Current, T* Buffer) : ResourceA(Current), ResourceB(Buffer)
105  {}
106 
107  /// @brief Make the buffered resource the active and vice versa.
108  void SwapUsableAndCommitable()
109  { std::swap(ResourceA,ResourceB); }
110 
111  /// @brief Get a pointer to the resource that can be used
112  /// @return A pointer to the usable resource
113  T* GetUsablePointer() const
114  { return ResourceA; }
115 
116  /// @brief Get a pointer to the resource that can be committed.
117  /// @return A pointer to the commitable resource
118  T* GetCommittablePointer() const
119  { return ResourceB; }
120 
121  /// @brief Get a reference to the resource that can be used
122  /// @return A reference to the usable resource
123  T& GetUsable()
124  { return *ResourceA; }
125 
126  /// @brief Get a reference to the resource that can be committed.
127  /// @return A reference to the commitable resource
128  T& GetCommittable()
129  { return *ResourceB; }
130 
131  /// @brief Destructor, this deletes both the resources managed. Regardless of how it got a hold of the pointers.
132  ~DoubleBufferedResource()
133  {
134  delete ResourceA;
135  delete ResourceB;
136  }
137  };//DoubleBufferedResource
