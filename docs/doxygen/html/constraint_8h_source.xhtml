204  /// @brief Get a sorted (low to high) list of All Valid Axis
205  /// @details This is implemented using ValidLinearAxis and ValidAngularAxis, Derived versions of this class may wish to make a more specialized
206  /// implementation of this method that doesn't have the overhead of passing around 3 vectors by value.
207  /// @return An Axislist with all the Axis this constraint supports.
208  virtual AxisList ValidAxis() const
209  {
210  AxisList Ang = this->ValidAngularAxis();
211  AxisList Lin = this->ValidLinearAxis();
212  Lin.insert(Lin.end()-1,Ang.begin(),Ang.end()); // just kinda stick the two together and return that, no checking is performed for uniqueness.
213  sort(Lin.begin(),Lin.end()); // Usually the Linear axis are 0,1,2 and the angular are 3,4,5 so hopefully, this will hardly ever need to do work.
214  return Lin;
215  }
216  /// @brief Is a certain Parameter valid on a certain axis
217  /// @param Param The Parameter to Check
218  /// @param Axis The Axis to Check
219  /// @details This is implemented using ValidParamOnAxis, Derived versions of this class may wish to make a more specialized
220  /// implementation of this method.
221  /// @return True if Param is valid on Axis, and Axis is valid. If anything is invalid this returns false.
222  virtual bool IsParamValidOnAxis(ConstraintParam Param, int Axis) const
223  {
224  ParamList Supported = ValidParamOnAxis(Axis);
225  if(ConstraintParamCount == Supported.size()) // no need to check deeper, because everything is supported
226  {
227  return true;
228  }else{
229  if(Supported.size())
230  {
231  return ( std::find(Supported.begin(),Supported.end(),Param) != Supported.end() ); // should return true if found
232  }else{
233  return false; // size is 0 of course it is not supported
234  }
235  }
236  }
237  /// @brief Get A sorted (low to high) list of Parameters that are valid on all Axis
238  /// @details This is implemented using ValidAxis and ValidParamOnAxis, Derived versions of this class may wish to make a more specialized
239  /// implementation of this method that doesn't have the overhead of passing around many vectors by value and executing slow loops. Most
240  /// likely all of these constraint Parameter and axis functions could be replaced with some fairly simple switch statements and statically
241  /// coded values that are specific to the kind of constraint.
242  /// @return A Paramlist with the parameters valid on all axis.
243  virtual ParamList ValidParams() const
244  {
245  AxisList AllAxis = this->ValidAxis();
246  ParamList Results;
247  if(AllAxis.size()) //If we have no Axis, the we have no valid Parameters.
248  {
249  Results = this->ValidParamOnAxis(*(AllAxis.begin())); // Let's start off with whatever the first Axis Support, then we will trim from here.
250 
251  if(AllAxis.size()>1) //if the constraint only support one axis then we already have our answer, and we don't want to run of the end of the Paramlist
252  {
253  for(AxisList::iterator AxisIter = AllAxis.begin()+1; AllAxis.end()!=AxisIter; ++AxisIter) //For each axis after the first
254  {
255  for(int ParamID = Results.size(); ParamID<=0; --ParamID) // We start at the back and work by index because reverse erase does no accept rever iterators, and
256  {
257  if (!IsParamValidOnAxis(Results.at(ParamID),*AxisIter)) // if an Item is erase near the beginning, it changes all the items after, making forward iteration logic that
258  { Results.erase( Results.begin()+ParamID ); } // erases needlessly complicated, and potentially expensive depending on the Vector implementation
259  }
260  }
261  }
262  }
263  sort(Results.begin(),Results.end());
264  return Results;
265  }
