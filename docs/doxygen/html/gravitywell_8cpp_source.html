97  void GravityWell::ApplyEffect()
98  {
99  /// @todo This currently will apply this fields force uniformly to all rigid proxies contained in a WorldObject.
100  /// Instead this should perhaps apply only to the ones in the field, or perhaps apply force based on the proxy position
101  /// rather than the WorldObject position to get more interesting results.
102  /// @todo Update to allow the application of force to soft proxies.
103 
104  if( 0 == this->Strength )
105  return;
106 
107  if( !this->AllowWorldGrav && !this->AddedObjects.empty() )
108  {
109  for( ObjectIterator AddedIt = this->AddedObjects.begin() ; AddedIt != this->AddedObjects.end() ; ++AddedIt )
110  {
111  ProxyContainer RigidProxies;
112  (*AddedIt)->GetProxies(Mezzanine::PT_Physics_RigidProxy,RigidProxies);
113  for( ProxyIterator ProxIt = RigidProxies.begin() ; ProxIt != RigidProxies.end() ; ++ProxIt )
114  {
115  Physics::RigidProxy* RigProx = static_cast<Physics::RigidProxy*>( *ProxIt );
116  RigProx->SetGravity( Vector3(0,0,0) );
117  }
118  }
119  }
120 
121  if( !this->OverlappingObjects.empty() ) {
122  const Vector3 SelfLoc = this->Ghost->GetLocation();
123  Vector3 ObjectLoc, Direction;
124  Real Distance = 0,AppliedStrength = 0;
125 
126  for( ObjectIterator ObjIt = this->OverlappingObjects.begin() ; ObjIt != this->OverlappingObjects.end() ; ObjIt++ )
127  {
128  ProxyContainer RigidProxies;
129  (*ObjIt)->GetProxies(Mezzanine::PT_Physics_RigidProxy,RigidProxies);
130  if( RigidProxies.empty() )
131  continue;
132 
133  Distance = ObjectLoc.Distance(SelfLoc);
134  Direction = (SelfLoc - ObjectLoc) / Distance;
135  switch(this->AttenStyle)
136  {
137  case Mezzanine::Att_Linear:
138  AppliedStrength = Strength - (AttenAmount * Distance);
139  break;
140  case Mezzanine::Att_Quadratic:
141  AppliedStrength = Strength - (AttenAmount * (Distance * Distance));
142  break;
143  default:
144  AppliedStrength = Strength;
145  break;
146  }
147 
148  //Apply the Force
149  for( ProxyIterator ProxIt = RigidProxies.begin() ; ProxIt != RigidProxies.end() ; ++ProxIt )
150  {
151  Physics::RigidProxy* RigProx = static_cast<Physics::RigidProxy*>( *ProxIt );
152 
153  Real Mass = RigProx->GetMass();
154  if( 0 > AppliedStrength ) {
155  AppliedStrength = 0;
156  }
157 
158  RigProx->ApplyForce( Direction * (AppliedStrength * Mass ) );
159  }
160  }
161  }
162 
163  if( !this->AllowWorldGrav && !this->RemovedObjects.empty() )
164  {
165  const Vector3 WorldGravity = Entresol::GetSingletonPtr()->GetPhysicsManager()->GetWorldGravity();
166  for( ObjectIterator RemovedIt = this->RemovedObjects.begin() ; RemovedIt != this->RemovedObjects.end() ; ++RemovedIt )
167  {
168  ProxyContainer RigidProxies;
169  (*RemovedIt)->GetProxies(Mezzanine::PT_Physics_RigidProxy,RigidProxies);
170  for( ProxyIterator ProxIt = RigidProxies.begin() ; ProxIt != RigidProxies.end() ; ++ProxIt )
171  {
172  Physics::RigidProxy* RigProx = static_cast<Physics::RigidProxy*>( *ProxIt );
173  RigProx->SetGravity( WorldGravity );
174  }
175  }
176  }
177  }

215  void GravityWell::ProtoSerializeProperties(XML::Node& SelfRoot) const
216  {
217  this->AreaEffect::ProtoSerializeProperties(SelfRoot);
218 
219  XML::Node PropertiesNode = SelfRoot.AppendChild( GravityWell::GetSerializableName() + "Properties" );
220 
221  if( PropertiesNode.AppendAttribute("Version").SetValue("1") &&
222  PropertiesNode.AppendAttribute("AttenAmount").SetValue( this->GetAttenuationAmount() ) &&
223  PropertiesNode.AppendAttribute("AttenStyle").SetValue( this->GetAttenuationStyle() ) &&
224  PropertiesNode.AppendAttribute("Strength").SetValue( this->GetFieldStrength() ) &&
225  PropertiesNode.AppendAttribute("AllowWorldGravity").SetValue( this->GetAllowWorldGravity() ? "true" : "false" ) )
226  {
227  return;
228  }else{
229  SerializeError("Create XML Attribute Values",GravityWell::GetSerializableName() + "Properties",true);
230  }
231  }

233  void GravityWell::ProtoDeSerializeProperties(const XML::Node& SelfRoot)
234  {
235  this->AreaEffect::ProtoDeSerializeProperties(SelfRoot);
236 
237  XML::Attribute CurrAttrib;
238  XML::Node PropertiesNode = SelfRoot.GetChild( GravityWell::GetSerializableName() + "Properties" );
239 
240  if( !PropertiesNode.Empty() ) {
241  if(PropertiesNode.GetAttribute("Version").AsInt() == 1) {
242  CurrAttrib = PropertiesNode.GetAttribute("AttenAmount");
243  if( !CurrAttrib.Empty() )
244  this->SetAttenuationAmount( CurrAttrib.AsReal() );
245 
246  CurrAttrib = PropertiesNode.GetAttribute("AttenStyle");
247  if( !CurrAttrib.Empty() )
248  this->SetAttenuationStyle( static_cast<Mezzanine::AttenuationStyle>( CurrAttrib.AsWhole() ) );
249 
250  CurrAttrib = PropertiesNode.GetAttribute("Strength");
251  if( !CurrAttrib.Empty() )
252  this->SetFieldStrength( CurrAttrib.AsReal() );
253 
254  CurrAttrib = PropertiesNode.GetAttribute("AllowWorldGravity");
255  if( !CurrAttrib.Empty() )
256  this->SetAllowWorldGravity( StringTools::ConvertToBool( CurrAttrib.AsString() ) );
257  }else{
258  MEZZ_EXCEPTION(Exception::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for " + (GravityWell::GetSerializableName() + "Properties" ) + ": Not Version 1.");
259  }
260  }else{
261  MEZZ_EXCEPTION(Exception::II_IDENTITY_NOT_FOUND_EXCEPTION,GravityWell::GetSerializableName() + "Properties" + " was not found in the provided XML node, which was expected.");
262  }
263  }
