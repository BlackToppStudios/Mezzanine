00268     {
00269         Ogre::MaterialPtr TheMaterial = Ogre::MaterialManager::getSingleton().getByName(MaterialName);
00270         String GroupName = TheMaterial->getGroup();
00271 
00272         Real RingAngle = (Ogre::Math::PI / Rings);
00273         Real SegAngle = (2 * Ogre::Math::PI / Segments);
00274         Whole VertIndex = 0;
00275 
00276         Ogre::ManualObject* sphere = new Ogre::ManualObject("TempMan");
00277         sphere->begin(MaterialName);
00278 
00279         for( Whole ring = 0 ; ring <= Rings ; ring++ ) {
00280         Real r0 = Radius * sinf (ring * RingAngle);
00281         Real y0 = Radius * cosf (ring * RingAngle);
00282 
00283             // Generate the group of segments for the current ring
00284             for( Whole seg = 0 ; seg <= Segments ; seg++ ) {
00285                 Real x0 = r0 * sinf(seg * SegAngle);
00286                 Real z0 = r0 * cosf(seg * SegAngle);
00287 
00288                 // Add one vertex to the strip which makes up the sphere
00289                 sphere->position( x0, y0, z0);
00290                 sphere->normal(Ogre::Vector3(x0, y0, z0).normalisedCopy());
00291                 sphere->textureCoord((Real) seg / (Real) Segments, (Real) ring / (Real) Rings);
00292 
00293                 if (ring != Rings) {
00294                     // each vertex (except the last) has six indicies pointing to it
00295                     sphere->index(VertIndex + Segments + 1);
00296                     sphere->index(VertIndex);
00297                     sphere->index(VertIndex + Segments);
00298                     sphere->index(VertIndex + Segments + 1);
00299                     sphere->index(VertIndex + 1);
00300                     sphere->index(VertIndex);
00301                     VertIndex++;
00302                 }
00303             }
00304         }
