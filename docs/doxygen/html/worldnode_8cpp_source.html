109  void WorldNode::SetDirection(const Vector3& Direction, const Mezzanine::TransformSpace& TS, const Vector3& LocalAxis)
110  {
111  static const Vector3 Zero(0,0,0);
112  if(Direction == Zero)
113  return;
114 
115  Vector3 NormalizedDir = Direction.GetNormal();
116  switch(TS)
117  {
118  default:
119  case Mezzanine::TS_World:
120  {
121  // Do nothing
122  break;
123  }
124  case Mezzanine::TS_Local:
125  {
126  NormalizedDir = GetOrientation() * NormalizedDir;
127  break;
128  }
129  case Mezzanine::TS_Parent:
130  {
131  if(Parent) NormalizedDir = Parent->GetOrientation() * NormalizedDir;
132  else return; /// @todo May want to change this to an exception, maybe.
133  break;
134  }
135  }
136 
137  Quaternion FinalOrientation;
138  if(FixedYaw)
139  {
140  Vector3 XVec = FixedYawAxis.CrossProduct(NormalizedDir);
141  XVec.Normalize();
142  Vector3 YVec = NormalizedDir.CrossProduct(XVec);
143  YVec.Normalize();
144  Quaternion ZToTarget(XVec,YVec,NormalizedDir);
145 
146  if(LocalAxis == Vector3::Neg_Unit_Z())
147  {
148  FinalOrientation.SetValues(-ZToTarget.Y,-ZToTarget.Z,ZToTarget.W,ZToTarget.X);
149  }else{
150  FinalOrientation = ZToTarget * (LocalAxis.GetRotationToAxis(Vector3::Unit_Z()));
151  }
152  }else{
153  Quaternion CurrOri = GetOrientation();
154  Vector3 CurrDir = CurrOri * LocalAxis;
155  if( (CurrDir+NormalizedDir).SquaredLength() < 0.00005 )
156  {
157  FinalOrientation.SetValues(-CurrOri.Y,-CurrOri.Z,CurrOri.W,CurrOri.X);
158  }else{
159  FinalOrientation = (CurrDir.GetRotationToAxis(NormalizedDir)) * CurrOri;
160  }
161  }
162 
163  SetOrientation(FinalOrientation);
164  }

362 Mezzanine::XML::Node& operator >> (const Mezzanine::XML::Node& OneNode, Mezzanine::WorldNode& Ev)
363 {
364  if ( Mezzanine::String(OneNode.Name())==Mezzanine::String("WorldNode") )
365  {
366  if(OneNode.GetAttribute("Version").AsInt() == 1)
367  {
368  Mezzanine::WorldNode * AttachPtr = Mezzanine::Entresol::GetSingletonPtr()->GetSceneManager()->GetNode( OneNode.GetAttribute("AttachedTo").AsString() );
369  if (AttachPtr)
370  { AttachPtr->AttachObject(&Ev); }
371 
372  Mezzanine::Vector3 TempVec(0,0,0);
373  Mezzanine::Quaternion TempQuat(0,0,0,0);
374  for(Mezzanine::XML::Node Child = OneNode.GetFirstChild(); Child!=0; Child = Child.GetNextSibling())
375  {
376  Mezzanine::String Name(Child.Name());
377  switch(Name[0])
378  {
379  case 'A':
380  if(Name==Mezzanine::String("Attached"))
381  {
382  Mezzanine::String AttributeName(OneNode.GetAttribute("Name").AsString());
383  Mezzanine::WorldNode * AttachPtr = Mezzanine::Entresol::GetSingletonPtr()->GetSceneManager()->GetNode( Child.GetAttribute("Name").AsString() );
384  /// @todo This doesn't account for other objects aside from world nodes to be attached. Additional checks that fetch other types by name in the case of a null pointer should be implemented.
385 
386  if (AttachPtr) // fail silently, because if we don't find it then that means it just hasn't been deserialized yeat
387  {
388  if (0==AttachPtr->GetParent())// unless it is attached to something else
389  {
390  Ev.AttachObject(AttachPtr);
391  }else{
392  MEZZ_EXCEPTION(Mezzanine::Exception::INVALID_STATE_EXCEPTION,"Cannot reconcile WorldNode with the current state of the world: Attachable \"" + AttachPtr->GetName() + "\" needs to be attached, but is already attached to " + AttachPtr->GetParent()->GetName() + ".");
393  }
394  }
395 
396  }else{
397  MEZZ_EXCEPTION(Mezzanine::Exception::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for WorldNode: Includes unknown Element D-\"" + Name + "\".");

399  break;
400  case 'O': //Orientation
