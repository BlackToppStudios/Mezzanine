473  void GravityWell::ApplyEffect()
474  {
475  if(0 == Strength)
476  return;
477  //ActorBase* Act = NULL;
478  ActorRigid* ActRig = NULL;
479  if(!AllowWorldGrav && !AddedActors.empty())
480  {
481  for ( std::vector<ActorBase*>::iterator AA = AddedActors.begin() ; AA != AddedActors.end() ; AA++ )
482  {
483  if(Mezzanine::WSO_ActorRigid != (*AA)->GetType())
484  continue;
485  ActRig = dynamic_cast<ActorRigid*>(*AA);
486  ActRig->GetPhysicsSettings()->SetIndividualGravity(Vector3());
487  }
488  }
489  if(!OverlappingActors.empty())
490  {
491  Vector3 ActorLoc, Direction;
492  Real Distance, AppliedStrength, InvMass;
493  Vector3 GhostLoc = this->GetLocation();
494  for ( std::list<ActorBase*>::iterator OA = OverlappingActors.begin() ; OA != OverlappingActors.end() ; OA++ )
495  {
496  if(Mezzanine::WSO_ActorRigid != (*OA)->GetType())
497  continue;
498  //Collect necessary data
499  ActorLoc = (*OA)->GetLocation();
500  Distance = ActorLoc.Distance(GhostLoc);
501  Direction = (GhostLoc - ActorLoc) / Distance;
502  switch(AttenStyle)
503  {
504  case Mezzanine::Att_Linear:
505  AppliedStrength = Strength - (AttenAmount * Distance);
506  break;
507  case Mezzanine::Att_Quadratic:
508  AppliedStrength = Strength - (AttenAmount * (Distance * Distance));
509  break;
510  default:
511  AppliedStrength = Strength;
512  break;
513  }
514  ActRig = static_cast<ActorRigid*>(*OA);
515  InvMass = ActRig->GetBulletObject()->getInvMass();
516  if(0 != InvMass)
517  AppliedStrength *= (1 / ActRig->GetBulletObject()->getInvMass());
518  else
519  AppliedStrength = 0;
520  if(0 > AppliedStrength)
521  AppliedStrength = 0;
522  //Apply the Force
523  ActRig->GetBulletObject()->applyCentralForce((Direction * AppliedStrength).GetBulletVector3());
524  }
525  }
526  if(!AllowWorldGrav && !RemovedActors.empty())
527  {
528  Vector3 WorldGrav = Entresol::GetSingletonPtr()->GetPhysicsManager()->GetGravity();
529  for ( std::vector<ActorBase*>::iterator RA = RemovedActors.begin() ; RA != RemovedActors.end() ; RA++ )
530  {
531  if(Mezzanine::WSO_ActorRigid != (*RA)->GetType())
532  continue;
533  ActRig = dynamic_cast<ActorRigid*>(*RA);
534  ActRig->GetPhysicsSettings()->SetIndividualGravity(WorldGrav);
535  }
536  }
537  }
