154  {

218  void CellGrid::RegenerateGrid(const Vector2& WinDim)
219  {
220  if(!GridDirty)
221  return;
222  ClearGrid();
223  Vector2 ActPos = GridBack->GetActualPosition();
224 
225  Whole CurrentColumn = 0;
226  Whole CurrentRow = 0;
227  Whole CurrentXposition = (Whole)ActPos.X + (Whole)(EdgeSpacing.X);
228  Whole CurrentYposition = (Whole)ActPos.Y + (Whole)(EdgeSpacing.Y);
229  Whole StartXposition = CurrentXposition;
230  Whole StartYposition = CurrentYposition;
231  Whole RowLimit = (Whole)((ActPos.X + WorkAreaLimits.X) - EdgeSpacing.X * 2);
232  Whole ColumnLimit = (Whole)((ActPos.Y + WorkAreaLimits.Y) - EdgeSpacing.Y * 2);
233 
234  UI::Cell* CurrCell = NULL;
235 
236  if( CellGrid::CG_Horizontal_Vertical_Ascending == Ordering || CellGrid::CG_Horizontal_Vertical_Decending == Ordering )
237  {
238  for( CellList::iterator it = Cells.begin() ; it != Cells.end() ; it++ )
239  {
240  CurrCell = (*it);
241  Vector2 CellSize = CurrCell->GetActualSize();
242  if(RowLimit >= CurrentXposition + (Whole)(CellSize.X + CellSpacing.X))
243  {
244  TheGrid.at(CurrentRow)->push_back(CurrCell);
245  CurrentXposition+=CellSize.X + CellSpacing.X;
246  CurrentColumn++;
247  }
248  else if(RowLimit >= CurrentXposition + (Whole)CellSize.X)
249  {
250  TheGrid.at(CurrentRow)->push_back(CurrCell);
251  CurrentXposition+=CellSize.X + CellSpacing.X;
252  CurrentColumn++;
253  }
254  else
255  {
256  if(ColumnLimit < CurrentYposition + (Whole)(CellSize.Y + CellSpacing.Y))
257  break;
258  CurrentXposition = StartXposition;
259  CurrentColumn = 0;
260  CurrentYposition+=(Whole)(FixedCellSize.Y + CellSpacing.Y);
261  CurrentRow++;
262  if(CurrentRow >= TheGrid.size())
263  CreateOrDestroyRow(0);
264 
265  TheGrid.at(CurrentRow)->push_back(CurrCell);
266  CurrentXposition+=CellSize.X + CellSpacing.X;
267  CurrentColumn++;
268  }
269  }
270  }
271  else if( CellGrid::CG_Vertical_Horizontal_Ascending == Ordering || CellGrid::CG_Vertical_Horizontal_Decending == Ordering )
272  {
273  for( CellList::iterator it = Cells.begin() ; it != Cells.end() ; it++ )
274  {
275  CurrCell = (*it);
276  Vector2 CellSize = CurrCell->GetActualSize();
277  if(ColumnLimit >= CurrentYposition + (Whole)(CellSize.Y + CellSpacing.Y))
278  {
279  TheGrid.at(CurrentRow)->push_back(CurrCell);
280  CurrentYposition+=CellSize.Y + CellSpacing.Y;
281  CurrentRow++;
282  if(CurrentRow >= TheGrid.size())
283  CreateOrDestroyRow(0);
284  }
285  else if(ColumnLimit >= CurrentYposition + (Whole)CellSize.Y)
286  {
287  TheGrid.at(CurrentRow)->push_back(CurrCell);
288  CurrentYposition+=CellSize.Y + CellSpacing.Y;
289  CurrentRow++;
290  if(CurrentRow >= TheGrid.size())
291  CreateOrDestroyRow(0);
292  }
293  else
294  {
295  if(RowLimit < CurrentXposition + (Whole)(CellSize.X + CellSpacing.X))
296  break;
297  CurrentYposition = StartYposition;
298  CurrentRow = 0;
299  CurrentXposition+=(Whole)(FixedCellSize.X + CellSpacing.X);
300  CurrentColumn++;
301 
302  TheGrid.at(CurrentRow)->push_back(CurrCell);
303  CurrentRow++;
304  if(CurrentRow >= TheGrid.size())
305  CreateOrDestroyRow(0);
306  }
307  }
308  }
309  GridDirty = false;
310  }
