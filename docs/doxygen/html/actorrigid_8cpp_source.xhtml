182  void ActorRigid::_NotifyCollisionState(Physics::Collision* Col, const Physics::Collision::CollisionState& State)
183  {
184  WorldObject::_NotifyCollisionState(Col,State);
185  StickyData* StickyD = GetPhysicsSettings()->GetStickyData();
186  bool UseA = Col->GetObjectA() != this;
187  // We don't care if sticky behavior isn't set or if the collision has ended.
188  // If it's ended, then we've probably already done our logic.
189  if(0 == StickyD->MaxNumContacts || Physics::Collision::Col_End == State)
190  return;
191  // Constraints only work well with other rigid bodies, so for now just other actorrigids.
192  /// @todo Update this to be workable with other objects that have rigid bodies internally, and maybe soft bodies.
193  if( (UseA ? (Col->GetObjectA()->GetType() != Mezzanine::WSO_ActorRigid) : (Col->GetObjectB()->GetType() != Mezzanine::WSO_ActorRigid)) )
194  return;
195  // We need a contact point to be present for this to work, since a collision without contact points is an AABB overlap.
196  // So confirm there are contact points.
197  if(Col->GetNumContactPoints() < 1)
198  return;
199  // Do we have room for additional sticky constraints?
200  if(StickyD->StickyConstraints.size() < StickyD->MaxNumContacts)
201  {
202  // Does this sticky constraint already exist for this pair?
203  for( Whole X = 0 ; X < StickyD->StickyConstraints.size() ; ++X )
204  {
205  StickyConstraint* StickyCon = StickyD->StickyConstraints.at(X);
206  if( Col->PairsMatch(StickyCon->GetActorA(),StickyCon->GetActorB()) )
207  return;
208  }
209  // Alternatively, is this sticky constraint about to be constructed?
210  for( Whole Y = 0 ; Y < StickyD->CreationQueue.size() ; ++Y )
211  {
212  StickyConstraintConstructionInfo& CurrInfo = StickyD->CreationQueue.at(Y);
213  if( Col->PairsMatch(CurrInfo.ActA,CurrInfo.ActB) )
214  return;
215  }
216  // Does this collision have a contact that is actually penetrating?
217  // Does this collision have actual force?
218  bool NegativeDistFound = false;
219  Whole FoundIndex = 0;
220  Real BestMatch = 123000;
221  for( Whole X = 0 ; X < Col->GetNumContactPoints() ; ++X )
222  {
223  Real Dist = Col->GetDistance(X);
224  if( MathTool::Fabs(Dist) < BestMatch && Dist <= 0 &&
225  0 != Col->GetAppliedImpulse(X) )
226  {
227  FoundIndex = X;
228  BestMatch = MathTool::Fabs(Dist);
229  NegativeDistFound = true;
230  break;
231  }
232  }
233  if(!NegativeDistFound)
234  return;
235  // Ok, it's passed all the tests, the constraint is being made.
236  // Get the actor and it's sticky data.
237  StickyConstraintConstructionInfo NewInfo;
238  NewInfo.ActA = dynamic_cast<ActorRigid*>(UseA ? Col->GetObjectA() : Col->GetObjectB());
239  NewInfo.ActB = this;
240  //StickyData* ActorAStickyData = ActorA->GetPhysicsSettings()->GetStickyData();
241  // Calculate the transforms.
242  Vector3 ALoc = UseA ? Col->GetLocalALocation(FoundIndex) : Col->GetLocalBLocation(FoundIndex);
243  Vector3 BLoc = UseA ? Col->GetLocalBLocation(FoundIndex) : Col->GetLocalALocation(FoundIndex);
244  NewInfo.TransA = Transform(ALoc,NewInfo.ActA->GetOrientation());
245  NewInfo.TransB = Transform(BLoc,this->GetOrientation());
246  StickyD->CreationQueue.push_back(NewInfo);
247 
248  /*ObjectPair CollisionPair(ActorA,this);
249  PhysicsManager::GetSingletonPtr()->RemoveCollision(PhysicsManager::GetSingletonPtr()->GetCollision(&CollisionPair));
250 
251  // Create and configure the constraint.
252  StickyConstraint* NewSticky = new StickyConstraint(ActorA,this,TransA,TransB);
253  NewSticky->SetUpperLinLimit(0.0);
254  NewSticky->SetUpperAngLimit(0.0);
255  NewSticky->SetLowerLinLimit(0.0);
256  NewSticky->SetLowerAngLimit(0.0);
257  // Add the constraint to the world and other necessary structures.
258  PhysicsManager::GetSingletonPtr()->GetPhysicsWorldPointer()->addConstraint(NewSticky->GetConstraintBase(),true);
259  StickyD->StickyConstraints.push_back(NewSticky);
260  ActorAStickyData->StickyConstraints.push_back(NewSticky);// */
261  }
