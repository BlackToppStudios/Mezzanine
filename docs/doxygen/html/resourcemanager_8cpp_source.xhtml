118  bool ResourceManager::CreateDirectory(const String& DirectoryPath)
119  {
120  #ifdef WINDOWS
121  if(::CreateDirectoryA(DirectoryPath.c_str(),NULL) < 0)
122  {
123  if(ERROR_ALREADY_EXISTS == ::GetLastError())
124  {
125  return false;
126  }
127  StringStream ExceptionStream;
128  ExceptionStream << "Unable to create directory. Error follows:" << std::endl;
129  if(ERROR_PATH_NOT_FOUND == ::GetLastError())
130  {
131  ExceptionStream << "Path to requested directory does not exist.";
132  }
133  else
134  {
135  ExceptionStream << "Error Unknown. :(";
136  }
137  MEZZ_EXCEPTION(Exception::IO_DIRECTORY_NOT_FOUND_EXCEPTION,ExceptionStream.str());
138  }
139  return true;
140  #else
141  if(::mkdir(DirectoryPath.c_str(),0777) < 0)
142  {
143  if( EEXIST == errno )
144  {
145  return false;
146  }
147  StringStream ExceptionStream;
148  ExceptionStream << "Unable to create directory. Error follows:" << std::endl;
149  ExceptionStream << strerror(errno);
150  MEZZ_EXCEPTION(Exception::IO_DIRECTORY_NOT_FOUND_EXCEPTION,ExceptionStream.str());
151  }
152  return true;
153  #endif
154  }

156  bool ResourceManager::CreateDirectoryPath(const String& DirectoryPath)
157  {
158  bool Result = true;
159  StringVector FolderNames;
160  CountedPtr<StringVector> FolderVec = StringTools::Split(DirectoryPath,"/\\");
161  size_t StartIndex = 0;
162  String PathAttempt;
163  #ifdef WINDOWS
164  char SysSlash = '\\';
165  #else
166  char SysSlash = '/';
167  #endif
168  #ifdef WINDOWS
169  // For windows and windows like machines, see if the first entry is a drive, because attempting to make a drive is silly.
170  if(FolderVec->at(0).find(':') != String::npos)
171  {
172  PathAttempt.append( FolderVec->at(0) );
173  PathAttempt.append( 1, SysSlash );
174  StartIndex++;
175  }
176  #else
177  PathAttempt.append( 1, SysSlash );
178  #endif
179  for( size_t VecIndex = StartIndex ; VecIndex < FolderVec->size() ; ++VecIndex )
180  {
181  PathAttempt.append( FolderVec->at(VecIndex) );
182  PathAttempt.append( 1, SysSlash );
183  Result = this->CreateDirectory( PathAttempt );
184  }
185  return Result;
186  }

337  {
338  /// @todo This entire method is a bit of a hack. When the resource system gets refactored it should go through our archives or whatever equivalent.

341  NamedDataStreamIterator StreamIt = this->NamedDataStreams.find(AssetName);
