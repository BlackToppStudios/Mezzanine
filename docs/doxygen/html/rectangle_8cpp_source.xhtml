153  void Rectangle::RotationTransform(Vector2& TopLeft, Vector2& TopRight, Vector2& BottomLeft, Vector2& BottomRight)
154  {
155  if(0 != RotAngle)
156  {
157  Real RotCos = MathTool::Cos(RotAngle);
158  Real RotSin = MathTool::Sin(RotAngle);
159 
160  if (!CustomCenter)
161  {
162  RotCenter.X = TopLeft.X + (BottomRight.X - TopLeft.X) / 2;
163  RotCenter.Y = TopLeft.Y + (BottomRight.Y - TopLeft.Y) / 2;
164  }
165 
166  TopLeft.X -= RotCenter.X;
167  TopLeft.Y -= RotCenter.Y;
168 
169  TopRight.X -= RotCenter.X;
170  TopRight.Y -= RotCenter.Y;
171 
172  BottomLeft.X -= RotCenter.X;
173  BottomLeft.Y -= RotCenter.Y;
174 
175  BottomRight.X -= RotCenter.X;
176  BottomRight.Y -= RotCenter.Y;
177 
178  Vector2 A2 = TopLeft, B2 = TopRight, C2 = BottomLeft, D2 = BottomRight;
179  TopLeft.X = A2.X * RotCos - A2.Y * RotSin;
180  TopLeft.Y = A2.X * RotSin + A2.Y * RotCos;
181 
182  TopRight.X = B2.X * RotCos - B2.Y * RotSin;
183  TopRight.Y = B2.X * RotSin + B2.Y * RotCos;
184 
185  BottomLeft.X = C2.X * RotCos - C2.Y * RotSin;
186  BottomLeft.Y = C2.X * RotSin + C2.Y * RotCos;
187 
188  BottomRight.X = D2.X * RotCos - D2.Y * RotSin;
189  BottomRight.Y = D2.X * RotSin + D2.Y * RotCos;
190 
191  TopLeft.X += RotCenter.X;
192  TopLeft.Y += RotCenter.Y;
193 
194  TopRight.X += RotCenter.X;
195  TopRight.Y += RotCenter.Y;
196 
197  BottomLeft.X += RotCenter.X;
198  BottomLeft.Y += RotCenter.Y;
199 
200  BottomRight.X += RotCenter.X;
201  BottomRight.Y += RotCenter.Y;
202  }

205  void Rectangle::SetSprite(Sprite* PSprite)
206  {
207  if(PSprite == NULL)
208  {
209  PriAtlas = ParentScreen->GetPrimaryAtlas();
210  UVs[0] = UVs[1] = UVs[2] = UVs[3] = ParentScreen->GetSolidUV(PriAtlas);
211  _MarkDirty();
212  }else{
213  PriAtlas = PSprite->Atlas;
214  Real TexelOffsetX = ParentScreen->GetTexelOffsetX();
215  Real TexelOffsetY = ParentScreen->GetTexelOffsetY();
216  if(TexelOffsetX) TexelOffsetX /= ParentScreen->GetTextureSize(PSprite->Atlas).X;
217  if(TexelOffsetY) TexelOffsetY /= ParentScreen->GetTextureSize(PSprite->Atlas).Y;
218  UVs[0].X = UVs[3].X = PSprite->UVLeft - TexelOffsetX;
219  UVs[0].Y = UVs[1].Y = PSprite->UVTop - TexelOffsetY;
220  UVs[1].X = UVs[2].X = PSprite->UVRight + TexelOffsetX;
221  UVs[2].Y = UVs[3].Y = PSprite->UVBottom + TexelOffsetY;
222  _MarkDirty();
223  }

326  void Rectangle::SetBackgroundGradient(const UI::Gradient& Grad, const ColourValue& ColourA, const ColourValue& ColourB)
327  {

329  {

332  BackgroundColours[2] = ColourB;
333  BackgroundColours[3] = ColourB;

335  else if (UI::Gradient_WestEast == Grad)
336  {
337  BackgroundColours[0] = ColourA;
338  BackgroundColours[3] = ColourA;
339  BackgroundColours[1] = ColourB;
340  BackgroundColours[2] = ColourB;
341  }
342  else if (UI::Gradient_Diagonal_1 == Grad)
343  {
344  ColourValue Average;
345  Average.R = (ColourA.R + ColourB.R) * 0.5f;
346  Average.G = (ColourA.G + ColourB.G) * 0.5f;
347  Average.B = (ColourA.B + ColourB.B) * 0.5f;
348  Average.A = (ColourA.A + ColourB.A) * 0.5f;
349  BackgroundColours[0] = ColourA;
350  BackgroundColours[1] = Average;
351  BackgroundColours[2] = ColourB;
352  BackgroundColours[3] = Average;
353  }
354  else if (UI::Gradient_Diagonal_2 == Grad)
355  {
356  ColourValue Average;
357  Average.R = (ColourA.R + ColourB.R) * 0.5f;
358  Average.G = (ColourA.G + ColourB.G) * 0.5f;
359  Average.B = (ColourA.B + ColourB.B) * 0.5f;
360  Average.A = (ColourA.A + ColourB.A) * 0.5f;
361  BackgroundColours[0] = Average;
362  BackgroundColours[1] = ColourA;

499  void Rectangle::_Redraw()
500  {
501  if(Dirty == false)
502  return;
503  RenderVertices.clear();
504  if(!Visible)
505  {
506  Dirty = false;
507  return;
508  }
509 
510  Real TexelOffsetX = ParentScreen->GetTexelOffsetX();
511  Real TexelOffsetY = ParentScreen->GetTexelOffsetY();
512  Vector2 TopLeft, TopRight, BottomLeft, BottomRight;
513  TopLeft.X = ActPosition.X + TexelOffsetX; TopLeft.Y = ActPosition.Y + TexelOffsetY;
514  TopRight.X = (ActPosition.X + ActSize.X) + TexelOffsetX; TopRight.Y = ActPosition.Y + TexelOffsetY;
515  BottomLeft.X = ActPosition.X + TexelOffsetX; BottomLeft.Y = (ActPosition.Y + ActSize.Y) + TexelOffsetY;
516  BottomRight.X = (ActPosition.X + ActSize.X) + TexelOffsetX; BottomRight.Y = (ActPosition.Y + ActSize.Y) + TexelOffsetY;
517  Vector2 OuterTopLeft = TopLeft, OuterTopRight = TopRight, OuterBottomLeft = BottomLeft, OuterBottomRight = BottomRight;
518 
519  // Rotation
520  RotationTransform(TopLeft,TopRight,BottomLeft,BottomRight);
521 
522  // Border
523  if(0.0 != BorderWidth)
524  {
525  OuterTopLeft.X -= BorderWidth; OuterTopLeft.Y -= BorderWidth;
526  OuterTopRight.X += BorderWidth; OuterTopRight.Y -= BorderWidth;
527  OuterBottomLeft.X -= BorderWidth; OuterBottomLeft.Y += BorderWidth;
528  OuterBottomRight.X += BorderWidth; OuterBottomRight.Y += BorderWidth;
529 
530  RotationTransform(OuterTopLeft,OuterTopRight,OuterBottomLeft,OuterBottomRight);
531  DrawBorder(TopLeft,TopRight,BottomLeft,BottomRight,
532  OuterTopLeft,OuterTopRight,OuterBottomLeft,OuterBottomRight);
533  }
534  // Fill
535  DrawFill(TopLeft,TopRight,BottomLeft,BottomRight);
536 
537  Dirty = false;
538  }
