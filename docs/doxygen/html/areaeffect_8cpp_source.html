86  void AreaEffect::CreateAreaEffect(Physics::CollisionShape* Shape)
87  {
88  Physics::PhysicsManager* PhysMan = Entresol::GetSingletonPtr()->GetPhysicsManager();
89  if( PhysMan != NULL ) {
90  if( Shape == NULL ) {
91  this->Ghost = PhysMan->CreateGhostProxy();
92  }else{
93  this->Ghost = PhysMan->CreateGhostProxy(Shape,false);
94  }
95 
96  UInt16 ColGroup = Physics::CF_SensorFilter;
97  UInt16 ColMask = Physics::CF_AllFilter & ~(Physics::CF_SensorFilter | Physics::CF_StaticFilter);
98  this->Ghost->SetCollisionGroupAndMask(ColGroup,ColMask);
99  this->Ghost->_Bind(this);
100  }
101  }

103  void AreaEffect::DestroyAreaEffect()
104  {
105  this->RemoveFromWorld();
106  Physics::PhysicsManager* PhysMan = Entresol::GetSingletonPtr()->GetPhysicsManager();
107  if( this->Ghost != NULL && PhysMan != NULL ) {
108  PhysMan->DestroyProxy( this->Ghost );
109  this->Ghost = NULL;
110  }

304  void AreaEffect::ProtoDeSerializeProxies(const XML::Node& SelfRoot)
305  {
306  this->DestroyAreaEffect();
307  // No base implementations to call
308  //XML::Attribute CurrAttrib;
309  XML::Node ProxiesNode = SelfRoot.GetChild( AreaEffect::GetSerializableName() + "Proxies" );
310 
311  if( !ProxiesNode.Empty() ) {
312  if(ProxiesNode.GetAttribute("Version").AsInt() == 1) {
313  /// @todo I don't think an exception is appropriate for the failure of the worldmanager validity checks,
314  /// however a warning should be written to the log if that happens. This should be updated to do that once
315  /// logging refactors are done.
316 
317  XML::Node GhostProxNode = ProxiesNode.GetChild("Ghost").GetFirstChild();
318  if( !GhostProxNode.Empty() ) {
319  Physics::PhysicsManager* PhysMan = Entresol::GetSingletonPtr()->GetPhysicsManager();
320  if( PhysMan ) {
321  this->Ghost = PhysMan->CreateGhostProxy(GhostProxNode);
322  this->Ghost->_Bind( this );
323  }
324  }
325  }else{
326  MEZZ_EXCEPTION(Exception::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for " + (AreaEffect::GetSerializableName() + "Proxies" ) + ": Not Version 1.");
327  }
328  }else{
329  MEZZ_EXCEPTION(Exception::II_IDENTITY_NOT_FOUND_EXCEPTION,AreaEffect::GetSerializableName() + "Proxies" + " was not found in the provided XML node, which was expected.");
330  }
331  }

342  void AreaEffect::_Update()
343  {
344  this->AddedObjects.clear();
345  this->RemovedObjects.clear();
346 
347  ObjectContainer PrevOverlapping;
348  PrevOverlapping.swap( this->OverlappingObjects );
349 
350  const UInt32 NumProxies = this->Ghost->GetNumShapeOverlappingProxies();
351  for( UInt32 ProxIndex = 0 ; ProxIndex < NumProxies ; ++ProxIndex )
352  {
353  Physics::CollidableProxy* CurrProxy = this->Ghost->GetShapeOverlappingProxy( ProxIndex );
354  if( CurrProxy != NULL ) {
355  WorldObject* CurrObject = CurrProxy->GetParentObject();
356  if( CurrObject != NULL ) {
357  // We need to check for unique world objects just in case a world object contains multiple collidable proxies
358  ObjectIterator UniqueCheck = std::find( this->OverlappingObjects.begin(), this->OverlappingObjects.end(), CurrObject );
359  if( UniqueCheck == this->OverlappingObjects.end() ) {
360  // We've established that this object is unique, so lets try to remove it from our temporary vector
361  ObjectIterator PrevCheck = std::find( PrevOverlapping.begin(), PrevOverlapping.end(), CurrObject );
362  if( PrevCheck == PrevOverlapping.end() ) {
363  // If we've failed to find it, then it's new
364  this->OverlappingObjects.push_back( CurrObject );
365  this->AddedObjects.push_back( CurrObject );
366  }else{
367  // If it was in the previous frame it isn't new, but it may be removed
368  this->OverlappingObjects.push_back( CurrObject );
369  PrevOverlapping.erase( PrevCheck );
370  }
371  }
372  }
373  }
374  }
375 
376  // If we have anything left over in our prev-overlapping, it's not inside this anymore so put them in the removed vector
377  if( !PrevOverlapping.empty() ) {
378  this->RemovedObjects.swap( PrevOverlapping );
379  }
380  }
