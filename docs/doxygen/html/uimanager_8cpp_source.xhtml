151  void UIManager::HotKeyAndInputCaptureChecks()
152  {
153  if(HotKeys.empty())
154  return;
155  std::vector<Input::InputCode> CapturedCodes;
156  std::list<EventUserInput*>* InputList = EventManager::GetSingletonPtr()->GetAllUserInputEvents();
157  Input::InputCode CurrCode;
158  for( std::list<EventUserInput*>::iterator Ilit = InputList->begin() ; Ilit != InputList->end() ; Ilit++ )
159  {
160  for( Whole X = 0 ; X < (*Ilit)->size() ; X++ )
161  {
162  CurrCode = (*Ilit)->GetMetaCode(X).GetCode();
163  if(LastWidgetSelected && LastWidgetSelected->IsInputCaptureWidget())
164  {
165  CapturedCodes.push_back(CurrCode);
166  }else{
167  HotKeyActivationCheck(CurrCode);
168  }
169  }
170  }
171  if(!CapturedCodes.empty())
172  LastWidgetSelected->GetInputCaptureData()->UpdateCapturedInputs(CapturedCodes);
173  }

175  void UIManager::WidgetUpdates()
176  {
177  if(HoveredWidget)
178  HoveredWidget->Update();
179  if(HoveredWidget || WidgetFocus)
180  {
181  Input::ButtonState State = Input::InputManager::GetSingletonPtr()->GetSystemMouse()->GetButtonState(1);
182  if(Input::BUTTON_PRESSING == State)
183  {
184  WidgetFocus = HoveredWidget;
185  }
186  else if(Input::BUTTON_DOWN == State)
187  {
188  if(HoveredWidget != WidgetFocus && WidgetFocus)
189  WidgetFocus->Update(true);
190  }
191  else if(Input::BUTTON_LIFTING == State)
192  {
193  if(HoveredWidget != WidgetFocus && WidgetFocus)
194  WidgetFocus->Update(true);
195  else if(HoveredWidget && HoveredWidget == WidgetFocus)
196  LastWidgetSelected = HoveredWidget;
197  WidgetFocus = NULL;
198  }
199  }
200  else if(!HoveredWidget && !WidgetFocus)
201  {
202  Input::ButtonState State = Input::InputManager::GetSingletonPtr()->GetSystemMouse()->GetButtonState(1);
203  if(LastWidgetSelected && Input::BUTTON_LIFTING == State)
204  LastWidgetSelected = NULL;
205  }

223  void UIManager::MouseActivationCheck(Button* ToCheck)
224  {
225  if(!ToCheck)
226  return;
227 
228  Input::Mouse* SysMouse = Input::InputManager::GetSingletonPtr()->GetSystemMouse();
229  const std::vector<Input::InputCode>& MouseCodes = ToCheck->GetMouseActivationButtons();
230  Input::ButtonState State = ToCheck->IsLiftActivation() ? Input::BUTTON_LIFTING : Input::BUTTON_PRESSING;
231 
232  for( Whole X = 0 ; X < MouseCodes.size() ; X++ )
233  {
234  Input::InputCode Code = MouseCodes.at(X);
235  if( SysMouse->CheckButtonState(Code,State) )
236  {
237  ToCheck->_SetActivation(true);
238  ActivatedButtons.push_back(ToCheck);
239  }
240  }

476  UIManager::GlyphIndexResult UIManager::SuggestGlyphIndex(const Whole& Height, const String& Atlas)
477  {
478  TextureAtlas* TheAtlas = NULL;
479  AtlasIterator It = Atlases.find(Atlas);
480  if(It != Atlases.end()) TheAtlas = (*It).second;
481  else
482  {
483  MEZZ_EXCEPTION(Exception::PARAMETERS_EXCEPTION,"Attempting to access TextureAtlas \"" + Atlas + "\", which does not exist or is not loaded.");
484  }
485  std::map<UInt32,GlyphData*>& Glyphs = TheAtlas->GetGlyphs();
486  Whole LargerMatch = 0;
487  Whole SmallerMatch = 0;
488  Real LargerHeight = 0;
489  Real SmallerHeight = 0;
490  Real LargerMatchDiff = 1000000.0;
491  Real SmallerMatchDiff = 1000000.0;
492  Real RequestedHeight = (Real)Height;
493 
494  for( std::map<UInt32,GlyphData*>::iterator it = Glyphs.begin() ; it != Glyphs.end() ; it++ )
495  {
496  Real Diff = 0.0;
497  if((*it).second->LineHeight > RequestedHeight)
498  {
499  Diff = (*it).second->LineHeight - RequestedHeight;
500  if(Diff < LargerMatchDiff)
501  {
502  LargerMatch = (*it).first;
503  LargerHeight = (*it).second->LineHeight;
504  LargerMatchDiff = Diff;
505  continue;
506  }
507  }else{
508  Diff = RequestedHeight - (*it).second->LineHeight;
509  if(Diff < SmallerMatchDiff)
510  {
511  SmallerMatch = (*it).first;
512  SmallerHeight = (*it).second->LineHeight;
513  SmallerMatchDiff = Diff;
514  continue;
515  }
516  }
517  }
518 
519  Real Scale = 1;
520  if(LargerMatch != 0)
521  {
522  Scale = RequestedHeight / LargerHeight;
523  return GlyphIndexResult(LargerMatch,Scale);
524  }else{
525  Scale = RequestedHeight / SmallerHeight;
526  return GlyphIndexResult(SmallerMatch,Scale);
527  }
528  }
