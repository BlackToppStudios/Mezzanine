92  AxisAlignedBox AxisAlignedBox::GetOverlap(const AxisAlignedBox& Other) const
93  {
94  Vector3 NewMin = this->MinExt, NewMax = this->MaxExt;
95  NewMin.Ceil(Other.MinExt);
96  NewMax.Floor(Other.MaxExt);
97 
98  if( NewMin.X < NewMax.X && NewMin.Y < NewMax.Y && NewMin.Z < NewMax.Z ) {
99  return AxisAlignedBox(NewMin,NewMax);
100  }else{
101  return AxisAlignedBox();
102  }
103  }
