211  void Collision::Update()
212  {
213  InternalAlgo->getAllContactManifolds(InternalData->Manifolds);
214  Whole NumManifolds = InternalData->Manifolds.size();
215  if( PenetrationDistances.size() != NumManifolds )
216  {
217  UpdatePenetrationDistances();
218  ObjectA->_NotifyCollisionState(this,Collision::Col_Contacts_Updated);
219  ObjectB->_NotifyCollisionState(this,Collision::Col_Contacts_Updated);
220  return;
221  }
222  for( Whole X = 0 ; X < NumManifolds ; ++X )
223  {
224  if( PenetrationDistances[X] != GetDistance(X) )
225  {
226  UpdatePenetrationDistances();
227  ObjectA->_NotifyCollisionState(this,Collision::Col_Contacts_Updated);
228  ObjectB->_NotifyCollisionState(this,Collision::Col_Contacts_Updated);
229  return;
230  }
231  }
