64  {

66  Real ActThick = Thickness * WinDim.X;
67 
68  HoriRect.Position = Vector2(RelPosition.X,(RelPosition.Y + RelSize.Y) - ActThick / WinDim.Y);

70  HoriRect.Relative = Rect.Relative;
71  VertRect.Position = Vector2((RelPosition.X + RelSize.X) - Thickness,RelPosition.Y);
72  VertRect.Size = Vector2(Thickness,RelSize.Y - (ActThick / WinDim.Y));
73  VertRect.Relative = Rect.Relative;
74  }else{
75  HoriRect.Position = Vector2(Rect.Position.X,(Rect.Position.Y + Rect.Size.Y) - Thickness);
76  HoriRect.Size = Vector2(Rect.Size.X - Thickness,Thickness);
77  HoriRect.Relative = Rect.Relative;
78  VertRect.Position = Vector2((Rect.Position.X + Rect.Size.X) - Thickness,Rect.Position.Y);
79  VertRect.Size = Vector2(Thickness,Rect.Size.Y - Thickness);
80  VertRect.Relative = Rect.Relative;
81  }
82 
83  HorizontalScroll = new UI::Scrollbar(Name+"HoriScr",HoriRect,Style,ParentScreen);
84  VerticalScroll = new UI::Scrollbar(Name+"VertScr",VertRect,Style,ParentScreen);
85  HorizontalScroll->Hide();
86  VerticalScroll->Hide();

132  void ScrolledCellGrid::DrawGrid(const Vector2& WinDim)
133  {
134  if(Cells.empty())
135  return;
136  RegenerateGrid();
137  for( Whole X = 0 ; X < VisibleCells.size() ; X++ )
138  VisibleCells[X]->Hide();
139  VisibleCells.clear();
140 
141  Whole CurrentRow = 0;
142  Whole CurrentColumn = 0;
143  if( GetMaxRows() > GetMaxYCellsPresentable() )
144  {
145  ScrollYHiding = false;
146  if(Visible && AutoHide)
147  VerticalScroll->Show();
148  CurrentRow = FindTopRow( VerticalScroll->GetScrollerValue() );
149  }else{
150  ScrollYHiding = true;
151  VerticalScroll->Hide();
152  }
153  if( GetMaxColumns() > GetMaxXCellsPresentable() )
154  {
155  ScrollXHiding = false;
156  if(Visible && AutoHide)
157  HorizontalScroll->Show();
158  CurrentColumn = FindLeftColumn( HorizontalScroll->GetScrollerValue() );
159  }else{
160  ScrollXHiding = true;
161  HorizontalScroll->Hide();
162  }
163 
164  const Whole StartRow = CurrentRow;
165  const Whole StartColumn = CurrentColumn;
166 
167  Vector2 ActPos = GetActualPosition();
168  Vector2 ActSize = GetActualSize();
169  Whole CursorXPos = (Whole)(ActPos.X + EdgeSpacing.X);
170  Whole CursorYPos = (Whole)(ActPos.Y + EdgeSpacing.Y);
171  const Whole StartXPos = CursorXPos;
172  const Whole StartYPos = CursorYPos;
173 
174  UI::Cell* CurrentCell = NULL;
175  bool StillHasRoom = true;
176 
177  switch (Ordering)
178  {
179  case CG_Horizontal_Vertical_Ascending:
180  case CG_Horizontal_Vertical_Decending:
181  {
182  if(ActSize.Y < (FixedCellSize.Y + (EdgeSpacing.Y * 2)))
183  return;
184  while(StillHasRoom)
185  {
186  CurrentCell = GetCell(CurrentRow,CurrentColumn);
187  if(!CurrentCell)
188  {
189  CursorYPos+=FixedCellSize.Y + CellSpacing.Y;
190  CurrentRow++;
191  CursorXPos = StartXPos;
192  CurrentColumn = StartColumn;
193  if(CurrentRow >= GetMaxRows())
194  StillHasRoom = false;
195  continue;
196  }
197  Vector2 CellSize = CurrentCell->GetActualSize();
198  if((Real)CursorXPos + CellSize.X > (ActPos.X + ActSize.X) - EdgeSpacing.X)
199  {
200  CursorYPos+=FixedCellSize.Y + CellSpacing.Y;
201  CurrentRow++;
202  CursorXPos = StartXPos;
203  CurrentColumn = StartColumn;
204  if(CursorYPos+FixedCellSize.Y > (ActPos.Y + ActSize.Y) - EdgeSpacing.Y)
205  StillHasRoom = false;
206  continue;
207  }
208  CurrentCell->SetActualPosition(Vector2((Real)CursorXPos,(Real)CursorYPos));
209  if(Visible)
210  CurrentCell->Show();
211  VisibleCells.push_back(CurrentCell);
212  CursorXPos+=(Whole)(CellSize.X + CellSpacing.X);
213  CurrentColumn++;
214  }
215  break;
216  }
217  case CG_Vertical_Horizontal_Ascending:
218  case CG_Vertical_Horizontal_Decending:
219  {
220  if(ActSize.X < (FixedCellSize.X + (EdgeSpacing.X * 2)))
221  return;
222  while(StillHasRoom)
223  {
224  CurrentCell = GetCell(CurrentRow,CurrentColumn);
225  if(!CurrentCell)
226  {
227  CursorXPos+=FixedCellSize.X + CellSpacing.X;
228  CurrentColumn++;
229  CursorYPos = StartYPos;
230  CurrentRow = StartRow;
231  if(CurrentColumn >= GetMaxColumns())
232  StillHasRoom = false;
233  continue;
234  }
235  Vector2 CellSize = CurrentCell->GetActualSize();
236  if((Real)CursorYPos + CellSize.Y > (ActPos.Y + ActSize.Y) - EdgeSpacing.Y)
237  {
238  CursorXPos+=FixedCellSize.X + CellSpacing.X;
239  CurrentColumn++;
240  CursorYPos = StartYPos;
241  CurrentRow = StartRow;
242  if(CursorXPos+FixedCellSize.X > (ActPos.X + ActSize.X) - EdgeSpacing.X)
243  StillHasRoom = false;
244  continue;
245  }
246  CurrentCell->SetActualPosition(Vector2((Real)CursorXPos,(Real)CursorYPos));
247  if(Visible)
248  CurrentCell->Show();
249  VisibleCells.push_back(CurrentCell);
250  CursorYPos+=(Whole)(CellSize.Y + CellSpacing.Y);
251  CurrentRow++;
252  }
253  break;
254  }
255  }
