70  void Viewport::SetCamera(CameraProxy* ViewportCamera)
71  {
72  Ogre::Camera* ViewCam = ViewportCamera ? ViewportCamera->_GetGraphicsObject() : NULL;
73  if(ViewportCam && ViewportCam != ViewportCamera)
74  ViewportCam->CameraVP = NULL;
75 
76  OgreViewport->setCamera(ViewCam);
77  this->ViewportCam = ViewportCamera;
78  if(ViewportCam)
