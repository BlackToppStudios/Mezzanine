124  String Timer::GetTimeAsText(const Timer::TimeFormat Format)
125  {
126  StringStream TimeStream;
127  switch( Format )
128  {
129  case Timer::TF_Raw_Micro:
130  {
131  TimeStream << this->GetCurrentTime();
132  return TimeStream.str();
133  break;
134  }
135  case Timer::TF_Raw_Milli:
136  {
137  TimeStream << this->GetCurrentTimeInMilliseconds();
138  return TimeStream.str();
139  break;
140  }
141  case Timer::TF_Seconds:
142  {
143  TimeStream << ( this->GetCurrentTimeInMilliseconds() / 1000 );
144  return TimeStream.str();
145  break;
146  }
147  case Timer::TF_Seconds_Milli:
148  {
149  Whole Milli = this->GetCurrentTimeInMilliseconds();
150  Whole Seconds = ( Milli / 1000 );
151  Whole Remainder = ( Milli % 1000 );
152  if( Remainder < 10 ) {
153  TimeStream << Seconds << ".00" << Remainder;
154  }else if( Remainder < 100 ) {
155  TimeStream << Seconds << ".0" << Remainder;
156  }else{
157  TimeStream << Seconds << "." << Remainder;
158  }
159  break;
160  }
161  case Timer::TF_Minutes_Seconds:
162  {
163  Whole Seconds = ( this->GetCurrentTimeInMilliseconds() / 1000 );
164  Whole Minutes = Seconds / 60;
165  Whole Remainder = Seconds % 60;
166  if( Remainder < 10 ) {
167  TimeStream << Minutes << ":0" << Remainder;
168  }else{
169  TimeStream << Minutes << ":" << Remainder;
170  }
171  return TimeStream.str();
172  break;
173  }
174  default:
175  {
176  return "";
177  break;
178  }
179  }
180  return "";
181  }

224  void StopWatchTimer::Update()
225  {
226  if( this->StartStamp != 0 ) {
227  MaxInt CurrentStamp = crossplatform::GetTimeStamp();
228  if( CurrentStamp - this->StartStamp > this->CurrentTime ) {
229  this->CurrentTime = 0;
230  }else{
231  this->CurrentTime -= ( CurrentStamp - this->StartStamp );
232  }
233  this->StartStamp = CurrentStamp;
234  }
235 
236  if( this->GoalReached() ) {
237  this->StartStamp = 0;
238  if( this->ResetAtGoal ) {
239  this->Reset();
240  }
241  }
242  }
