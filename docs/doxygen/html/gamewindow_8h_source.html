87  WF_FSAA_16 = 64, ///< Enables Fullscreen Anti-Aliasing level 16 for the window.
88  WF_Resizeable = 128, ///< Creates a window with resizable borders, otherwise it is fixed size.
89  WF_Maximized = 256, ///< Maximizes the window immediately after construction.
