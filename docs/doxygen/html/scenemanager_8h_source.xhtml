92  enum SceneShadowTechnique
93  { //Shadow Docs from Ogre ShadowTechnique Documentation.
94  SST_None = 0, ///< No shadows.
95  SST_Stencil_Modulative = 1, ///< Stencil shadow technique which renders all shadow volumes as a modulation after all the non-transparent areas have been rendered.
96  SST_Stencil_Additive = 2, ///< Stencil shadow technique which renders each light as a separate additive pass to the scene.
97  SST_Texture_Modulative = 11, ///< Texture-based shadow technique which involves a monochrome render-to-texture of the shadow caster and a projection of that texture onto the shadow receivers as a modulative pass.
98  SST_Texture_Additive = 12, ///< Texture-based shadow technique which involves a render-to-texture of the shadow caster and a projection of that texture onto the shadow receivers, built up per light as additive passes.
99  SST_Texture_Additive_Integrated = 13, ///< Texture-based shadow technique which involves a render-to-texture of the shadow caster and a projection of that texture on to the shadow receivers, with the usage of those shadow textures completely controlled by the materials of the receivers.
100  SST_Texture_Modulative_Integrated = 14 ///< Texture-based shadow technique which involves a render-to-texture of the shadow caster and a projection of that texture on to the shadow receivers, with the usage of those shadow textures completely controlled by the materials of the receivers.
101  };

185  /// @brief Creates a skyplane for use in making a sky.
186  /// @details Only one skyplane can exist in a scene. Making a new one will remove the old one. Skyplanes are
187  /// flat planes that face in one direction. They are ideal for levels with surrounding mountains or anything
188  /// where the horizon is not visable.
189  /// @param SkyPlane The plane that will become the sky.
190  /// @param Material The name of the material to be applied to the skyplane. Note: This is not referring to the
191  /// filename, but the specific material script within the file.
192  /// @param Group The resource group where the material can be found.
193  /// @param Scale The scaling to be applied to the skyplane. This may need to be tweaked based on how high you
194  /// set the plane off the ground.
195  /// @param Tiling The number of times to tile the texture or textures listed in the material script across the skyplane.
196  /// @param DrawFirst Whether or not the skyplane should be the first thing rendered in the scene. Usually you will
197  /// want this to be true as it'll ensure all other objects are rendered on top of it.
198  /// @param Bow This will add curvature to the skyplane if set above zero. Note: Use small numbers. A bow of 1.5
199  /// should be noticable.
200  /// @param XSegments The number of segments, or boxes, the skyplane consists of on the planes X axis. This is usful
201  /// when giving the skyplane a bow. By default the skyplane is just one massive box.
202  /// @param YSegments The number of segments, or boxes, the skyplane consists of on the planes Y axis. This is usful
203  /// when giving the skyplane a bow. By default the skyplane is just one massive box.
204  void CreateSkyPlane(const Plane& SkyPlane_, const String& Material, const String& Group, Real Scale=1000.0, Real Tiling=10.0,
205  bool DrawFirst=true, Real Bow=0, int XSegments=1, int YSegments=1);
206  /// @brief Disables the currently active skyplane.
207  /// @details Using this function effectively deletes the skyplane, so you will have to provide a new set of parameters
208  /// if you wish to re-create the skyplane.
209  void DisableSkyPlane();
210 
211  /// @brief Creates a skybox for use in making a sky.
212  /// @details Like skyplanes, only one can exist per scene. Unlike skyplanes, skyboxes will be applied individually to
213  /// each camera in the scene. The skybox will move with the camera, so as a result the camera will never be able to
214  /// "touch" the sky. Skyboxes are more performance intensive then skyplanes.
215  /// @param Material The name of the material to be applied to the skybox. Note: This is not referring to the
216  /// filename, but the specific material script within the file.
217  /// @param Group The resource group where the material can be found.
218  /// @param Distance The distance from the camera where the skybox is found. This is in world units.
219  /// @param DrawFirst Whether or not the skybox should be the first thing rendered in the scene. Usually you will
220  /// want this to be true as it'll ensure all other objects are rendered on top of it.
221  /// @param Orientation Optional quaternion to rotate the orientation of the skybox.
222  void CreateSkyBox(const String& Material, const String& Group, Real Distance, bool DrawFirst=true, Quaternion Orientation=Quaternion());
223  /// @brief Disables the currently active skybox.
224  /// @details Using this function effectively deletes the skybox, so you will have to provide a new set of parameters
225  /// if you wish to re-create the skybox.
226  void DisableSkyBox();

228  /// @brief Creates a skydome for use in making a sky.
229  /// @details Like the other two types of sky's, their can be only one skydome per scene. Skydomes much like skyboxes, except
230  /// they have 5 sides(the bottom side is missing), and they bow each of the sides to make the dome. In all other respects they
231  /// are the same.
232  /// @param Material The name of the material to be applied to the skydome. Note: This is not referring to the
233  /// filename, but the specific material script within the file.
234  /// @param Group The resource group where the material can be found.
235  /// @param Distance The distance from the camera where the skydome is found. This is in world units.
236  /// @param Curvature Curvature of the dome. Usually you want this value to be between 2 and 65.
237  /// @param Tiling The number of times to tile the texture or textures listed in the material script across the skydome.
238  /// @param DrawFirst Whether or not the skybox should be the first thing rendered in the scene. Usually you will
239  /// want this to be true as it'll ensure all other objects are rendered on top of it.
240  /// @param Orientation Optional quaternion to rotate the orientation of the skydome.
241  /// @param XSegments The number of segments, or boxes, the skydome consists of on the dome's X axis.
242  /// @param YSegments The number of segments, or boxes, the skydome consists of on the dome's Y axis.
243  void CreateSkyDome(const String& Material, const String& Group, Real Distance, Real Curvature=10.0, Real Tiling=8.0, bool DrawFirst=true,
244  Quaternion Orientation=Quaternion(), int XSegments=16, int YSegments=16);
245  /// @brief Disables the currently active skydome.
