57  /// @enum ExceptionCodes
58  /// @brief This enum provides mapping/naming for all the core exceptions of Mezzanine.
59  /// @details The numbers listed here must be unique to each exception. If an error code does not identify
60  /// the problem exactly it should be considered 'Unknown' for as much as it does specify. For example,

62  /// @ref IO_EXCEPTION would make sense

67  IO_FILE_READ_EXCEPTION = 0x01F01F01, ///< When used a instance of FileReadException is thrown. Thrown when a file could not be read and permissions seem to not be an issue (Filesystem/hardware issue?)

70  IO_FILE_PERMISSION_EXCEPTION = 0x01F01F04, ///< When used a instance of FilePermissionException is thrown. Thrown when permission was denied to a file.
71  IO_DIRECTORY_EXCEPTION = 0x01F02F00, ///< When used a instance of DirectoryException is thrown. Thrown when there is an unknown issue with a file.
72  IO_DIRECTORY_READ_EXCEPTION = 0x01F02F01, ///< When used a instance of DirectoryReadException is thrown. Thrown when a directory could be read and it wasn't a permission issue.
73  IO_DIRECTORY_WRITE_EXCEPTION = 0x01F02F02, ///< When used a instance of DirectoryWriteException is thrown. Thrown when a directory could be written to and it wasn't a permission issue.
74  IO_DIRECTORY_NOT_FOUND_EXCEPTION = 0x01F02F03, ///< When used a instance of DirectoryNotFoundException is thrown. Thrown when a directory was expected to be there, but was not.
75  IO_DIRECTORY_PERMISSION_EXCEPTION = 0x01F02F04, ///< When used a instance of DirectoryPermissionException is thrown. Thrown when permission is denied to a directory.
76  IO_NETWORK_EXCEPTION = 0x01F03F00, ///< When used a instance of NetworkException is thrown. Thrown when something unknown causes network IO to fail.
77  IO_NETWORK_READ_EXCEPTION = 0x01F03F01, ///< When used a instance of NetworkReadException is thrown. Thrown when data could not be read from the network (downloads).
78  IO_NETWORK_WRITE_EXCEPTION = 0x01F03F02, ///< When used a instance of NetworkWriteException is thrown. Thrown when data could not be read from the network (iloads).
79  IO_NETWORK_NOT_FOUND_EXCEPTION = 0x01F03F03, ///< When used a instance of NetworkNotFoundException is thrown. Thrown when no network connection is available.
80  IO_NETWORK_URL_EXCEPTION = 0x01F03F04, ///< When used a instance of NetworkURLException is thrown. Thrown when an address is invalid or could not be found.
81  IO_NETWORK_PERMISSION_EXCEPTION = 0x01F03F05, ///< When used a instance of NetworkPermissionException is thrown. Thrown when permision was denied to a network interface or network resource.
82  IO_WRITE_EXCEPTION = 0x01F08F00, ///< When used a instance of IOWriteException is thrown. Thrown when a write is happening but something has prevented the underlying code from knowing what was writing.
83  IO_READ_EXCEPTION = 0x01F09F00, ///< When used a instance of IOReadException is thrown. Thrown when a read is happening but something has prevented the underlying code from knowing what was reading.
84 
85  II_EXCEPTION = 0x02F00F00, ///< When used a instance of InstanceIdentityException is thrown. Thrown when an unknown error with using an Identifier and it is invalid.
86  II_IDENTITY_INVALID_EXCEPTION = 0x02F01F00, ///< When used a instance of InstanceIdentityInvalidException is thrown. Thrown when the identity string wasn't valid at all.
87  II_IDENTITY_NOT_FOUND_EXCEPTION = 0x02F02F00, ///< When used a instance of InstanceIdentityNotFoundException is thrown. Thrown when the requested identity could not be found.
88  II_DUPLICATE_IDENTITY_EXCEPTION = 0x02F03F00, ///< When used a instance of InstanceIdentityDuplicateException is thrown. Thrown when duplicates of teh same identity string exist.
89 
90  MM_EXCEPTION = 0x03F00F00, ///< When used a instance of MemoryManagementException is thrown. Thrown when an unknown memory management exception occurs.
91  MM_OUT_OF_MEMORY_EXCEPTION = 0x03F01F00, ///< When used a instance of OutOfMemoryException is thrown. Thrown when A memory allocation was attempted and failed.
92  MM_OUT_OF_BOUNDS_EXCEPTION = 0x03F02F00, ///< When used a instance of MemoryOutOfBoundsException is thrown. Thrown when attempted to access something that really should note be accessed.
93 
94  SYNTAX_ERROR_EXCEPTION = 0x04F00F00, ///< When used a instance of SyntaxErrorException is thrown. Thrown when some kind of syntax exception
95  SYNTAX_ERROR_EXCEPTION_XML = 0x04F01F00, ///< When used a instance of SyntaxErrorXMLException is thrown. Thrown when and XML document is being parsed but is invalid
96  SYNTAX_ERROR_EXCEPTION_XPATH = 0x04F02F00, ///< When used a instance of SyntaxErrorXPathException is thrown. Thrown when an XPath query is being parsed but is invalid
97  SYNTAX_ERROR_EXCEPTION_LUA = 0x04F03F00, ///< When used a instance of SyntaxErrorLuaException is thrown. Thrown when lua code in incorrect.
98 
99  SCRIPT_EXCEPTION = 0x05F00F00, ///< When used a instance of ScriptException is thrown. Thrown when an unknown error happens with a script.
100  SCRIPT_EXCEPTION_LUA = 0x05F01F00, ///< When used a instance of ScriptLuaException is thrown. Thrown when an unknown error happens in a Lua script.
101  SCRIPT_EXCEPTION_LUA_YIELD = 0x05F01F01, ///< When used a instance of ScriptLuaYieldException is thrown. Thrown when Lua returns a yield and it should not have.
102  SCRIPT_EXCEPTION_LUA_RUNTIME = 0x05F01F02, ///< When used a instance of ScriptLuaRuntimeException is thrown. Thrown when a Lua script has a runtime error.
103  SCRIPT_EXCEPTION_LUA_ERRERR = 0x05F01F03, ///< When used a instance of ScriptLuaErrErrException is thrown. Thrown when Lua has an error handling an error.
104 
105  PARAMETERS_EXCEPTION = 0x06F01F00, ///< When used a instance of ParametersException is thrown. Thrown when parameters are checked at runtime and found invalid.
106  PARAMETERS_CAST_EXCEPTION = 0x06F01F01, ///< When used a instance of ParametersCastException is thrown. Thrown when a pointer parameter is checked at runtime and cannot be cast as expected.
107  PARAMETERS_RANGE_EXCEPTION = 0x06F01F02, ///< When used a instance of ParametersRangeException is thrown. Thrown when a passed parameter is checked at runtime and not in the expected range.
108 
109  ARITHMETIC_EXCEPTION = 0x00F01F00, ///< When used a instance of ArithmeticException is thrown. Thrown when Math has failed.
110  INVALID_VERSION_EXCEPTION = 0x00F02F00, ///< When used a instance of InvalidVersionException is thrown. Thrown when a version is accessed/parsed/required and it cannot work correctly or is missing.
111  INVALID_STATE_EXCEPTION = 0x00F03F00, ///< When used a instance of InvalidStateException is thrown. Thrown when the available information should have worked but failed for unknown reasons.
112  RENDERINGAPI_EXCEPTION = 0x00F04F00, ///< When used a instance of RenderingAPIException is thrown. Thrown when the graphics card/DirectX/OpenGL fail.
113  RT_ASSERTION_EXCEPTION = 0x00F05F00, ///< When used a instance of RuntimeAssertionException is thrown. Thrown when a rutime assertion could have been thrown.
114  INTERNAL_EXCEPTION = 0x00F06F00, ///< When used a instance of InternalException is thrown. Thrown when an unknown internal error occurred.
115  NOT_IMPLEMENTED_EXCEPTION = 0x00F07F00, ///< When used a instance of NotImplementedException is thrown. Thrown when we just have not coded a thing yet, but we knew what the API should look like.
116  INVALID_ASSIGNMENT = 0x00F08F00 ///< When used a instance of InvalidAssignment is thrown. Then when a complex class is assigned to itself or other invalid assignments occur.
117  };

175  /// @struct ExceptionFactory
176  /// @headerfile exception.h
177  /// @brief Template class that serves as the base for exception factories.
178  /// @details Additional exceptions and their factories have to specialize from this template changing the type
179  /// value to the new exception type. This allows our exception macro to find the appropriate factory and be
180  /// extendable with additional exceptions. Attempting to create an unknown exception simply won't compile on
181  /// account of the base exception class being abstract.
182  ///////////////////////////////////////////////////////////////////////////////
183  template <Whole N>
184  struct MEZZ_LIB ExceptionFactory
185  {
186  typedef Exception Type;
187  };//ExceptionFactory
