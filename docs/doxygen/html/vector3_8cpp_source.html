389  Quaternion Vector3::GetRotationToAxis(const Vector3& Axis, const Vector3& FallBackAxis) const
390  {
391  Quaternion Ret;
392  Vector3 Vec1 = *this;
393  Vector3 Vec2 = Axis;
394  Vec1.Normalize();
395  Vec2.Normalize();
396 
397  Real Dot = Vec1.DotProduct(Vec2);
398  if( Dot >= 1.0 )
399  {
400  return Ret;
401  }
402  if( Dot < (1e-6 - 1.0) )
403  {
404  if( FallBackAxis != Vector3() )
405  {
406  Ret.SetFromAxisAngle(MathTools::GetPi(),FallBackAxis);
407  }else{
408  Vector3 CrossAxis = Vector3::Unit_X().CrossProduct(*this);
409  if(CrossAxis.IsZeroLength())
410  CrossAxis = Vector3::Unit_Y().CrossProduct(*this);
411  CrossAxis.Normalize();
412  Ret.SetFromAxisAngle(MathTools::GetPi(),CrossAxis);
413  }
414  }else{
415  Real Sqr = MathTools::Sqrt( (1+Dot)*2 );
416  Real InvSqr = 1 / Sqr;
417 
418  Vector3 Cross = Vec1.CrossProduct(Vec2);
419 
420  Ret.X = Cross.X * InvSqr;
421  Ret.Y = Cross.Y * InvSqr;
422  Ret.Z = Cross.Z * InvSqr;
423  Ret.W = Sqr * 0.5f;
424  Ret.Normalize();
425  }
426  return Ret;
427  }

498  void Vector3::ProtoSerialize(XML::Node& CurrentRoot) const
499  {
500  Mezzanine::XML::Node VecNode = CurrentRoot.AppendChild(SerializableName());
501 
502  if(VecNode)
503  {
504  Mezzanine::XML::Attribute VersionAttr = VecNode.AppendAttribute("Version");
505  Mezzanine::XML::Attribute XAttr = VecNode.AppendAttribute("X");
506  Mezzanine::XML::Attribute YAttr = VecNode.AppendAttribute("Y");
507  Mezzanine::XML::Attribute ZAttr = VecNode.AppendAttribute("Z");
508  if( VersionAttr && XAttr && YAttr && ZAttr )
509  {
510  if( VersionAttr.SetValue("1") && XAttr.SetValue(this->X) && YAttr.SetValue(this->Y) && ZAttr.SetValue(this->Z))
511  {
512  return;
513  }else{
514  SerializeError("Create XML Attribute Values", SerializableName(),true);
515  }
516  }else{
517  SerializeError("Create XML Attributes", SerializableName(),true);
518  }
