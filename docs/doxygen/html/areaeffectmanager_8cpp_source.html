302  {
303  AreaEffectIterator AEIt = ( Index < this->GetNumAreaEffects() ? this->AreaEffects.begin() + Index : this->AreaEffects.end() );

417  void AreaEffectManager::Initialize()
418  {
419  if( !this->Initialized )
420  {
421  //WorldManager::Initialize();
422 
423  this->TheEntresol->GetScheduler().AddWorkUnitMain( this->AreaEffectUpdateWork, "AreaEffectUpdateWork" );
424 
425  Physics::PhysicsManager* PhysicsMan = this->TheEntresol->GetPhysicsManager();
426  if( PhysicsMan ) {
427  this->AreaEffectUpdateWork->AddDependency( PhysicsMan->GetSimulationWork() );
428  this->AreaEffectUpdateWork->AddDependency( PhysicsMan->GetWorldTriggerUpdateWork() );
429  }
430 
431  Mezzanine::ActorManager* ActorMan = this->TheEntresol->GetActorManager();
432  if( ActorMan ) {
433  this->AreaEffectUpdateWork->AddDependency( ActorMan->GetActorUpdateWork() );
434  }
435 
436  this->Initialized = true;
437  }
438  }
