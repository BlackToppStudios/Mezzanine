172  {
173  XML::Node G6dofSpringNode = CurrentRoot.AppendChild(SerializableName()); // The base node all the base constraint stuff will go in

177  Mezzanine::XML::Attribute Version = G6dofSpringNode.AppendAttribute("Version"); // Version
178  if (!Version)
179  { SerializeError("Create Version", SerializableName()); }
180  Version.SetValue(1);

182  this->Generic6DofConstraint::ProtoSerialize(G6dofSpringNode); // Serialize the 6dof, dualtransform and the constraint
183 
184  XML::Node SpringStiffness = G6dofSpringNode.AppendChild("SpringStiffness");
185  if (!SpringStiffness)
186  { SerializeError("Create SpringStiffness", SerializableName()); }
187  XML::Node SpringDamping = G6dofSpringNode.AppendChild("SpringDamping");
188  if (!SpringDamping)
189  { SerializeError("Create SpringDamping", SerializableName()); }
190  XML::Node SpringEnabled = G6dofSpringNode.AppendChild("SpringEnabled");
191  if (!SpringEnabled)
192  { SerializeError("Create SpringEnabled", SerializableName()); }
193 
194 
195  for (int c=0; c<6; ++c) // Each of the spring attributes
196  {
197  String AttrName("Axis"+StringTools::ConvertToString(c));
198 
199  Mezzanine::XML::Attribute AxisStiffness = SpringStiffness.AppendAttribute(AttrName);
200  if (!AxisStiffness)
201  { SerializeError("Create AxisStiffness-"+AttrName, SerializableName()); }
202  AxisStiffness.SetValue(this->GetSpringStiffness(c));
203 
204  Mezzanine::XML::Attribute AxisDamping = SpringDamping.AppendAttribute(AttrName);
205  if (!AxisDamping)
206  { SerializeError("Create AxisDamping-"+AttrName, SerializableName()); }
207  AxisDamping.SetValue(this->GetSpringDamping(c));
208 
209  Mezzanine::XML::Attribute AxisEnabled = SpringEnabled.AppendAttribute(AttrName);
210  if (!AxisEnabled)
211  { SerializeError("Create AxisEnabled-"+AttrName, SerializableName()); }
212  AxisEnabled.SetValue(this->GetSpringEnabled(c));
213  }

216  void Generic6DofSpringConstraint::ProtoDeSerialize(const XML::Node& OneNode)
217  {
218  if ( Mezzanine::String(OneNode.Name())==this->Generic6DofSpringConstraint::SerializableName() )
219  {
220  if(OneNode.GetAttribute("Version").AsInt() == 1)
221  {
222  XML::Node g6dof = OneNode.GetChild("Generic6DofConstraint");
223  if(!g6dof)
224  { DeSerializeError("locate Generic6DofConstraint node",SerializableName()); }
225  this->Generic6DofConstraint::ProtoDeSerialize(g6dof);
226 
227  XML::Node SpringStiffness = OneNode.GetChild("SpringStiffness");
228  if (!SpringStiffness)
229  { DeSerializeError("Find SpringStiffness Node", SerializableName()); }
230  XML::Node SpringDamping = OneNode.GetChild("SpringDamping");
231  if (!SpringDamping)
232  { DeSerializeError("Find SpringDamping Node", SerializableName()); }
233  XML::Node SpringEnabled = OneNode.GetChild("SpringEnabled");
234  if (!SpringEnabled)
235  { DeSerializeError("Find SpringEnabled Node", SerializableName()); }
236 
237  XML::Attribute SpringStiffnessAttr = SpringStiffness.GetFirstAttribute();
238  while(SpringStiffnessAttr)
239  {
240  String CurrentName(SpringStiffnessAttr.Name());
241  if (4>CurrentName.size())
242  { DeSerializeError("Invalid axis attribute", SerializableName()); }
243  this->SetSpringStiffness(char4ToAxis(CurrentName[4]),SpringStiffnessAttr.AsReal());
244  SpringStiffnessAttr = SpringStiffnessAttr.GetNextAttribute();
245  }
246 
247  XML::Attribute SpringDampingAttr = SpringDamping.GetFirstAttribute();
248  while(SpringDampingAttr)
249  {
250  String CurrentName(SpringDampingAttr.Name());
251  if (4>CurrentName.size())
252  { DeSerializeError("Invalid axis attribute", SerializableName()); }
253  this->SetSpringDamping(char4ToAxis(CurrentName[4]),SpringDampingAttr.AsReal());
254  SpringDampingAttr = SpringDampingAttr.GetNextAttribute();
255  }
256 
257  XML::Attribute SpringEnabledAttr = SpringEnabled.GetFirstAttribute();
258  while(SpringEnabledAttr)
259  {
260  String CurrentName(SpringEnabledAttr.Name());
261  if (4>CurrentName.size())
262  { DeSerializeError("Invalid axis attribute", SerializableName()); }
263  this->SetSpringEnabled(char4ToAxis(CurrentName[4]),SpringEnabledAttr.AsBool());
264  SpringEnabledAttr = SpringEnabledAttr.GetNextAttribute();
265  }
266  }else{
267  DeSerializeError("find usable serialization version",SerializableName());
268  }
269  }else{
270  DeSerializeError(String("find correct class to deserialize, found a ")+OneNode.Name(),SerializableName());
271  }
272  }
