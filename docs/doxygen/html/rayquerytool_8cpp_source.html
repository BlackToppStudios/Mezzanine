177  Boole RayQueryTool::GetFirstObjectOnRayByPolygon(Ray ObjectRay, Whole ObjectFlags)
178  {
179  ManagedRayQuery RayQuery;
180  Ogre::Ray Ooray = ObjectRay.GetOgreRay();
181 
182  if(!ExecuteQuery(RayQuery, Ooray))
183  { return ClearReturns(); }
184 
185  // at this point we have raycast to a series of different objects bounding boxes.
186  // we need to test these different objects to see which is the first polygon hit.
187  // there are some minor optimizations (distance based) that mean we wont have to
188  // check all of the objects most of the time, but the worst case scenario is that
189  // we need to test every triangle of every object.
190  Ogre::Real closest_distance = -1.0f;
191  Vector3 closest_result;
192  IntersectedObject=NULL;
193  Ogre::RaySceneQueryResult &query_result = RayQuery->getLastResults();
194  for (size_t qr_idx = 0; qr_idx < query_result.size(); qr_idx++)
195  {
196  // stop checking if we have found a raycast hit that is closer than all remaining entities
197  if ( (0.0f <= closest_distance) && (closest_distance < query_result[qr_idx].distance))
198  { break; }
199 
200  // only check this result if its a hit against an entity
201  if ((NULL != query_result[qr_idx].movable) && (0 == query_result[qr_idx].movable->getMovableType().compare("Entity")))
202  {
203  // get the entity to check
204  Ogre::Entity *pentity = static_cast<Ogre::Entity*>(query_result[qr_idx].movable);
205 
206  try
207  {
208  WorldObject* HitMetaInfo = Ogre::any_cast<Graphics::RenderableProxy*>(pentity->getUserAny())->GetParentObject();
209  if( HitMetaInfo && ( HitMetaInfo->GetType() & ObjectFlags ) )
210  {
211  // mesh data to retrieve
212  size_t vertex_count;
213  size_t index_count;
214  Ogre::Vector3 *vertices;
215  unsigned long *indices;
216 
217  // get the mesh information
218  Internal::MeshTools::GetMeshInformation( pentity, vertex_count, vertices, index_count, indices,
219  pentity->getParentNode()->_getDerivedPosition(),
220  pentity->getParentNode()->_getDerivedOrientation(),
221  pentity->getParentNode()->_getDerivedScale());
222 
223  // test for hitting individual triangles on the mesh
224  Boole new_closest_found = false;
225  for (size_t i = 0; i < index_count; i += 3)
226  {
227  // check for a hit against this triangle
228  std::pair<bool, Ogre::Real> hit = Ogre::Math::intersects(Ooray, vertices[indices[i]], vertices[indices[i+1]], vertices[indices[i+2]], true, false);
229 
230  // if it was a hit check if its the closest
231  if (hit.first && ((0.0f > closest_distance) || (hit.second < closest_distance)) )
232  {
233  closest_distance = hit.second; // this is the closest so far, save it off
234  new_closest_found = true;
235  }
236  }
237 
238  // free the vertices and indices memory
239  delete[] vertices;
240  delete[] indices;
241 
242  // if we found a new closest raycast for this object, update the closest_result before moving on to the next object.
243  if (new_closest_found)
244  {
245  closest_result = Ooray.getPoint(closest_distance);
246  IntersectedObject = Ogre::any_cast<Graphics::RenderableProxy*>(pentity->getUserAny())->GetParentObject();
247  }
248 
249  } // \if WSO_ActorRigid
250  }catch(...){
251  ClearReturns();
252  MEZZ_EXCEPTION(Exception::INTERNAL_EXCEPTION,"Failed during cast in actor raycast.");
253  }
254  } // \if entity
255  } // \if qr_idx
256 
257  // Change the closest point into a point relative to the Actor
258  if (IntersectedObject)
259  {
260  Offset = IntersectedObject->GetOrientation() * ((closest_result - IntersectedObject->GetLocation()) * IntersectedObject->GetScale());
261  ValidResult=true;
262  return ValidResult;
263  }else{
264  return ClearReturns();
265  }
266  }
