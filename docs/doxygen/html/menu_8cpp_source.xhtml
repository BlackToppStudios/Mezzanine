79  void Menu::UpdateImpl(bool Force)
80  {
81  Input::ButtonState State = Input::InputManager::GetSingletonPtr()->GetSystemMouse()->GetButtonState(1);
82  if(HoveredSubWidget)
83  {
84  Button* button = NULL;
85  Widget* SubSubWidget = HoveredSubWidget->GetHoveredSubWidget();
86  if( SubSubWidget && (Widget::W_Button == SubSubWidget->GetType()) )
87  button = static_cast<Button*>(SubSubWidget);
88  if(button && State == Input::BUTTON_LIFTING)
89  {
90  UI::MenuWindow* MenWin = static_cast<UI::MenuWindow*>(HoveredSubWidget);
91  UI::MenuWindow* ChildMenWin = MenWin->GetWindowOfAccessButton(button);
92  if(ChildMenWin)
93  {
94  if(!ChildMenWin->IsVisible())
95  {
96  if(MenuStack.back()->GetAutoHide())
97  MenuStack.back()->Hide();
98  MenuStack.push_back(ChildMenWin);
99  AddSubRenderable(SubRenderables.size(),ChildMenWin);
100  ChildMenWin->Show();
101  return;
102  }else{
103  RollMenuBackToWindow(MenWin);
104  }
105  }
106  else if(button == MenWin->GetBackButton())
107  {
108  MenWin->Hide();
109  MenuStack.pop_back();
110  SubRenderables.pop_back();
111  if(!MenuStack.back()->IsVisible())
112  MenuStack.back()->Show();
113  return;
114  }
115  }
116  }

155  void Menu::RollMenuBackToWindow(UI::MenuWindow* Win)
156  {
157  for( std::vector<UI::MenuWindow*>::reverse_iterator rit = MenuStack.rbegin() ; rit != MenuStack.rend() ; rit++ )
158  {
159  if(!(*rit)->IsVisible())
160  (*rit)->Show();
161  if( RootWindow == (*rit) )
162  break;
163  if( Win != (*rit))
164  {
165  (*rit)->Hide();
166  MenuStack.pop_back();
167  SubRenderables.pop_back();
168  }else{
169  break;
170  }
