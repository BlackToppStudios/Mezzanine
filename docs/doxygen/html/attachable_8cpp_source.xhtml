219  void AttachableChild::_RecalculateGlobalTransform(bool FromParent)
220  {
221  // Set data for parent updates appropriately.
222  // Can't assign directly in case "FromParent" is false and we need an update, such as when SetLocation() is invoked directly.
223  if(FromParent)
224  GlobalTransformDirty = true;
225 
226  if(Parent && GlobalTransformDirty)
227  {
228  this->SetOrientation( Parent->ConvertLocalToGlobal(LocalXform.Rotation) );
229  this->SetLocation( Parent->ConvertLocalToGlobal(LocalXform.Location) );
230  }
231  GlobalTransformDirty = false;
232  }
233 
234  void AttachableChild::_RecalculateLocalTransform()
235  {
236  if(Parent && LocalTransformDirty)
237  {
238  this->LocalXform.Location = Parent->ConvertGlobalToLocal( this->GetLocation() );
239  this->LocalXform.Rotation = Parent->ConvertGlobalToLocal( this->GetOrientation() );
240  }
241  LocalTransformDirty = false;
242  }
