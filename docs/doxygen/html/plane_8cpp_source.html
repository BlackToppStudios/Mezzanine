96  void Plane::Define(const Vector3& First, const Vector3& Second, const Vector3& Third)
97  {
98  Vector3 Edge1 = Second - First;
99  Vector3 Edge2 = Third - First;
100  this->Normal = Edge1.CrossProduct(Edge2);
101  this->Normal.Normalize();
102  this->Distance = -(this->Normal.DotProduct(First));
103  }

117  Plane::Side Plane::GetSide(const Vector3& Center, const Vector3& HalfSize) const
118  {
119  Real CenterDist = this->GetDistance(Center);
120  Real MaxDist = MathTools::Fabs( this->Normal.DotProduct(HalfSize) );
121  if( CenterDist < -MaxDist ) {

157  void Plane::ProtoSerialize(XML::Node& ParentNode) const
158  {
159  XML::Node SelfRoot = ParentNode.AppendChild( Plane::GetSerializableName() );
160 
161  if( SelfRoot.AppendAttribute("Version").SetValue("1") &&
162  SelfRoot.AppendAttribute("Distance").SetValue( this->Distance ) )
163  {
164  XML::Node CenterNode = SelfRoot.AppendChild("Normal");
165  this->Normal.ProtoSerialize( CenterNode );
166 
167  return;
168  }else{
169  SerializeError("Create XML Attribute Values",Plane::GetSerializableName(),true);
170  }
171  }

173  void Plane::ProtoDeSerialize(const XML::Node& SelfRoot)
174  {
175  XML::Attribute CurrAttrib;
176 
177  if( String(SelfRoot.Name()) == Plane::GetSerializableName() ) {
178  if(SelfRoot.GetAttribute("Version").AsInt() == 1) {
179  CurrAttrib = SelfRoot.GetAttribute("Distance");
180  if( !CurrAttrib.Empty() )
181  this->Distance = CurrAttrib.AsReal();
182 
183  // Get the properties that need their own nodes
184  XML::Node NormalNode = SelfRoot.GetChild("Normal").GetFirstChild();
185  if( !NormalNode.Empty() )
186  this->Normal.ProtoDeSerialize(NormalNode);
187  }else{
188  MEZZ_EXCEPTION(Exception::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for " + Plane::GetSerializableName() + ": Not Version 1.");
189  }
190  }else{
191  MEZZ_EXCEPTION(Exception::II_IDENTITY_NOT_FOUND_EXCEPTION,Plane::GetSerializableName() + " was not found in the provided XML node, which was expected.");
192  }
193  }
