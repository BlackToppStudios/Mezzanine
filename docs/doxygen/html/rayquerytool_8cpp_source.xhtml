84  Vector3WActor* RayQueryTool::GetFirstActorOnRayByPolygon(Ray ActorRay, Whole ObjectFlags)
85  {
86  VerifyRayQuery();
87  Ogre::Ray Ooray = ActorRay.GetOgreRay();
88 
89  if(NULL != RayQuery) //Double check that the Rayquery is valid
90  {
91  RayQuery->setRay(Ooray);
92  //RayQuery->setSortByDistance(true);
93  RayQuery->setQueryMask(-1);
94  if( RayQuery->execute().size() <= 0 ) //Did we hit anything
95  {
96  return NULL;
97  }
98  }else{ //Whoopsie something Failed
99  MEZZ_EXCEPTION(Exception::PARAMETERS_EXCEPTION,"Attempting to run a query on Null RaySceneQuery");
100  }
101 
102  // at this point we have raycast to a series of different objects bounding boxes.
103  // we need to test these different objects to see which is the first polygon hit.
104  // there are some minor optimizations (distance based) that mean we wont have to
105  // check all of the objects most of the time, but the worst case scenario is that
106  // we need to test every triangle of every object.
107  Ogre::Real closest_distance = -1.0f;
108  Vector3 closest_result;
109  Ogre::RaySceneQueryResult &query_result = RayQuery->getLastResults();
110  Vector3WActor* ClosestActor = new Vector3WActor();
111  for (size_t qr_idx = 0; qr_idx < query_result.size(); qr_idx++)
112  {
113  // stop checking if we have found a raycast hit that is closer than all remaining entities
114  if ( (0.0f <= closest_distance) && (closest_distance < query_result[qr_idx].distance))
115  { break; }
116 
117  // only check this result if its a hit against an entity
118  if ((NULL != query_result[qr_idx].movable) && (0 == query_result[qr_idx].movable->getMovableType().compare("Entity")))
119  {
120  // get the entity to check
121  Ogre::Entity *pentity = static_cast<Ogre::Entity*>(query_result[qr_idx].movable);
122 
123  try
124  {
125  WorldObject* HitMetaInfo = Ogre::any_cast<WorldObject*>(pentity->getUserAny());
126  if(HitMetaInfo->GetType() & ObjectFlags)
127  {
128  // mesh data to retrieve
129  size_t vertex_count;
130  size_t index_count;
131  Ogre::Vector3 *vertices;
132  unsigned long *indices;
133 
134  // get the mesh information
135  GetMeshInformation(pentity, vertex_count, vertices, index_count, indices,
136  pentity->getParentNode()->_getDerivedPosition(),
137  pentity->getParentNode()->_getDerivedOrientation(),
138  pentity->getParentNode()->_getDerivedScale());
139 
140  // test for hitting individual triangles on the mesh
141  bool new_closest_found = false;
142  for (size_t i = 0; i < index_count; i += 3)
143  {
144  // check for a hit against this triangle
145  std::pair<bool, Ogre::Real> hit = Ogre::Math::intersects(Ooray, vertices[indices[i]], vertices[indices[i+1]], vertices[indices[i+2]], true, false);
146 
147  // if it was a hit check if its the closest
148  if (hit.first && ((0.0f > closest_distance) || (hit.second < closest_distance)) )
149  {
150  closest_distance = hit.second; // this is the closest so far, save it off
151  new_closest_found = true;
152  }
153  }
154 
155  // free the vertices and indices memory
156  delete[] vertices;
157  delete[] indices;
158 
159  // if we found a new closest raycast for this object, update the closest_result before moving on to the next object.
160  if (new_closest_found)
161  {
162  closest_result = Ooray.getPoint(closest_distance);
163  WorldObject* WO = Ogre::any_cast<WorldObject*>(pentity->getUserAny());
164  ClosestActor->Actor = static_cast<ActorBase*>( WO );
165  }
166 
167  } // \if WSO_ActorRigid
168  }catch(...){
169  MEZZ_EXCEPTION(Exception::INTERNAL_EXCEPTION,"Failed during cast in actor raycast.");
170  }
171  } // \if entity
172  } // \if qr_idx
173 
174  //Change the closest point into a point relative to the Actor
175  if (ClosestActor->Actor != NULL)
176  //{ ClosestActor->Vector = closest_result - ClosestActor->Actor->GetLocation(); }
177  { ClosestActor->Vector = ClosestActor->Actor->GetOrientation() * ((closest_result - ClosestActor->Actor->GetLocation()) * ClosestActor->Actor->GetScaling()); }
178  return ClosestActor;
179  }

232  Vector3* RayQueryTool::RayPlaneIntersection(const Ray &QueryRay, const Plane &QueryPlane)
233  {
234  VerifyRayQuery();
235  try{
236  Vector3 u = QueryRay.To - QueryRay.From;
237  Vector3 p0 = Vector3(0,0,0);
238 
239  if(QueryPlane.Normal.X == 0 && QueryPlane.Normal.Y == 0 && QueryPlane.Normal.Z == 0)
240  {
241  return 0;
242  }
243  else{
244  if(QueryPlane.Normal.X != 0)
245  {
246  p0 = Vector3(QueryPlane.Distance,0,0);
247  }
248  else if(QueryPlane.Normal.Y != 0)
249  {
250  p0 = Vector3(0,QueryPlane.Distance,0);
251  }
252  else
253  {
254  p0 = Vector3(0,0,QueryPlane.Distance);
255  }
256  }
257 
258  Vector3 w = QueryRay.From - p0;
259 
260  Real D = u.DotProduct(QueryPlane.Normal);
261  Real N = -1 * w.DotProduct(QueryPlane.Normal);
262 
263  Real SMALL_NUM = 0.00000001;
264 
265  if( (D<0? -D : D) < SMALL_NUM) //Checks if the Plane behind the RAy
266  {
267  if(N == 0)
268  {
269  return new Vector3(QueryRay.From);
270  }
271  else
272  {
273  return 0;
274  }
275  }
276 
277  Real sI = N/D;
278 
279  if(sI < 0 || sI > 1) //checks if the ray is too long
280  {
281  return 0;
282  }
283 
284  Vector3 test = Vector3(QueryRay.From + (u * sI));
285 
286  Vector3* return_vector = new Vector3(QueryRay.From + (u * sI));
287 
288  Real distance = return_vector->Distance(QueryRay.From);
289 
290  if(distance > QueryRay.From.Distance(QueryRay.To))
291  {
292  return 0;
293  }
294 
295  return return_vector;
296  } catch(exception e) {
297  //In case we divide b
298  Entresol::GetSingletonPtr()->Log("WorldQueryTool Error:Failed while calculating Ray/Plane Intersection, Assuming no valid intersection. Error follows:");
299  Entresol::GetSingletonPtr()->Log(e.what());
300  return 0;
301  }

331  void RayQueryTool::GetMeshInformation( Ogre::Entity *entity,
332  size_t &vertex_count,
333  Ogre::Vector3* &vertices,
334  size_t &index_count,
335  unsigned long* &indices,
336  const Ogre::Vector3 &position,
337  const Ogre::Quaternion &orient,
338  const Ogre::Vector3 &scale)
339  {
340  bool added_shared = false;
341  size_t current_offset = 0;
342  size_t shared_offset = 0;
343  size_t next_offset = 0;
344  size_t index_offset = 0;
345  vertex_count = index_count = 0;
346 
347  Ogre::MeshPtr mesh = entity->getMesh();
348 
349 
350  bool useSoftwareBlendingVertices = entity->hasSkeleton();
351 
352  if (useSoftwareBlendingVertices)
353  { //10,000th line of code
354  entity->_updateAnimation();
355  }
356 
357  // Calculate how many vertices and indices we're going to need
358  for (unsigned short i = 0; i < mesh->getNumSubMeshes(); ++i)
359  {
360  Ogre::SubMesh* submesh = mesh->getSubMesh( i );
361 
362  // We only need to add the shared vertices once
363  if(submesh->useSharedVertices)
364  {
365  if( !added_shared )
366  {
367  vertex_count += mesh->sharedVertexData->vertexCount;
368  added_shared = true;
369  }
370  }
371  else
372  {
373  vertex_count += submesh->vertexData->vertexCount;
374  }
375 
376  // Add the indices
377  index_count += submesh->indexData->indexCount;
378  }
379 
380 
381  // Allocate space for the vertices and indices
382  vertices = new Ogre::Vector3[vertex_count];
383  indices = new unsigned long[index_count];
384 
385  added_shared = false;
386 
387  // Run through the submeshes again, adding the data into the arrays
388  for ( unsigned short i = 0; i < mesh->getNumSubMeshes(); ++i)
389  {
390  Ogre::SubMesh* submesh = mesh->getSubMesh(i);
391 
392  //Ogre::VertexData* vertex_data = submesh->useSharedVertices ? mesh->sharedVertexData : submesh->vertexData;
393  Ogre::VertexData* vertex_data;
394 
395  //When there is animation:
396  if(useSoftwareBlendingVertices)
397  {
398  // #ifdef BUILD_AGAINST_AZATHOTH
399  // vertex_data = submesh->useSharedVertices ? entity->_getSharedBlendedVertexData() : entity->getSubEntity(i)->_getBlendedVertexData();
400  // #else
401  vertex_data = submesh->useSharedVertices ? entity->_getSkelAnimVertexData() : entity->getSubEntity(i)->_getSkelAnimVertexData();
402  // #endif
403  }else{
404  vertex_data = submesh->useSharedVertices ? mesh->sharedVertexData : submesh->vertexData;
405  }
406 
407  if((!submesh->useSharedVertices)||(submesh->useSharedVertices && !added_shared))
408  {
409  if(submesh->useSharedVertices)
410  {
411  added_shared = true;
412  shared_offset = current_offset;
413  }
414 
415  const Ogre::VertexElement* posElem = vertex_data->vertexDeclaration->findElementBySemantic(Ogre::VES_POSITION);
416  Ogre::HardwareVertexBufferSharedPtr vbuf = vertex_data->vertexBufferBinding->getBuffer(posElem->getSource());
417  unsigned char* vertex = static_cast<unsigned char*>(vbuf->lock(Ogre::HardwareBuffer::HBL_READ_ONLY));
418 
419  // There is _no_ baseVertexPointerToElement() which takes an Ogre::Real or a double
420  // as second argument. So make it float, to avoid trouble when Ogre::Real will
421  // be comiled/typedefed as double:
422  // Ogre::Real* pReal;
423  float* pReal;
424 
425  for( size_t j = 0; j < vertex_data->vertexCount; ++j, vertex += vbuf->getVertexSize())
426  {
427  posElem->baseVertexPointerToElement(vertex, &pReal);
428  Ogre::Vector3 pt(pReal[0], pReal[1], pReal[2]);
429  vertices[current_offset + j] = (orient * (pt * scale)) + position;
430  }
431 
432  vbuf->unlock();
433  next_offset += vertex_data->vertexCount;
434  }
435 
436 
437  Ogre::IndexData* index_data = submesh->indexData;
438  size_t numTris = index_data->indexCount / 3;
439  Ogre::HardwareIndexBufferSharedPtr ibuf = index_data->indexBuffer;
440 
441  bool use32bitindexes = (ibuf->getType() == Ogre::HardwareIndexBuffer::IT_32BIT);
442 
443  unsigned long* pLong = static_cast<unsigned long*>(ibuf->lock(Ogre::HardwareBuffer::HBL_READ_ONLY));
444  unsigned short* pShort = reinterpret_cast<unsigned short*>(pLong);
445 
446 
447  size_t offset = (submesh->useSharedVertices)? shared_offset : current_offset;
448  size_t index_start = index_data->indexStart;
449  size_t last_index = numTris*3 + index_start;
450 
451  if (use32bitindexes)
452  {
453  for (size_t k = index_start; k < last_index; ++k)
454  { indices[index_offset++] = pLong[k] + static_cast<unsigned long>( offset ); }
455  }else{
456  for (size_t k = index_start; k < last_index; ++k)
457  { indices[ index_offset++ ] = static_cast<unsigned long>( pShort[k] ) + static_cast<unsigned long>( offset ); }
458  }
459 
460  ibuf->unlock();
461  current_offset = next_offset;
462  }
463  }
