62  void HotKeyHandler::BindHotKey(const Input::MetaCode& HotKey, UI::Button* BoundButton)
63  {
64  if( Input::KEY_FIRST < HotKey.GetCode() && Input::KEY_LAST > HotKey.GetCode() )

68  void HotKeyHandler::UnbindHotKey(const Input::MetaCode& HotKey, UI::Button* BoundButton)
69  {
70  std::pair<const HotKeyIterator,const HotKeyIterator> Result = HotKeys.equal_range(HotKey.GetCode());
71  for( HotKeyIterator It = Result.first ; It != Result.second ; )
72  {
73  if(BoundButton == (*It).second) {
74  this->HotKeys.erase(It++);
75  return;
76  }else{
77  ++It;
78  }
79  }
80  }
