220  void Vector2::ProtoSerialize(XML::Node& CurrentRoot) const
221  {
222  Mezzanine::XML::Node VecNode = CurrentRoot.AppendChild(SerializableName());
223  VecNode.SetName(SerializableName());
224 
225  Mezzanine::XML::Attribute VersionAttr = VecNode.AppendAttribute("Version");
226  Mezzanine::XML::Attribute XAttr = VecNode.AppendAttribute("X");
227  Mezzanine::XML::Attribute YAttr = VecNode.AppendAttribute("Y");
228  if( VersionAttr && XAttr && YAttr )
229  {
230  if( VersionAttr.SetValue("1") && XAttr.SetValue(this->X) && YAttr.SetValue(this->Y) )
231  {
232  return;
233  }else{
234  SerializeError("Create XML Attribute Values", SerializableName(),true);
235  }
236  }else{
237  SerializeError("Create XML Attributes", SerializableName(),true);
238  }
