98  /// @brief A RollingAverage that stores a copy of each record. and does the math when queried
99  template <typename RecordType> class BufferedRollingAverage : public RollingAverage<RecordType>
100  {
101  protected:
102  /// @brief The collection of all the records that are being have been added going back as far as the capacity will allow.
103  /// @note All writes to this are performed through the iterator @ref Current
104  std::vector<RecordType> Records;
105 
106  /// @brief The iterator used to treat an std::vector as a circular buffer.
107  typename std::vector<RecordType>::iterator Current;
108 
109  /// @brief Move Current to the next place to be written

122  BufferedRollingAverage(const Whole& RecordCount = MEZZ_FRAMESTOTRACK)

140  /// @brief Add up all the records and divide by the count. A simple arithmetic mean.
141  /// @return This returns the current average of all the records as a RecordType. If it is an Integer type floating parts are truncated.
142  virtual RecordType GetAverage() const
143  {
144  RecordType Results(0);
145  for(typename std::vector<RecordType>::const_iterator Iter = Records.begin(); Iter!=Records.end(); ++Iter)
146  {Results += *Iter;}
147  return Results/Records.size();
148  }

203  /// @brief A weighted average that does math with every insertion and stores nothing.
204  /// @details This is possibly an optimization over the BufferedRollingAverage. It has not yet been shown whether the amount of memory for

236  /// @brief Update the currently stored Rolling average with a new data point/record.
237  /// @param Datum Update the Current Average according to the following formula CurrentAverage = CurrentAverage * ((RecordCount-1)/RecordCount) + Datum/RecordCount.
238  virtual void Insert(RecordType Datum)
239  {
240  LastEntry = Datum;
241  CurrentAverage = ( MathType(this->CurrentAverage) ? MathType(this->CurrentAverage) : MathType(1)) // A zero really screws with averages that tend to move away from zero
242  * ((MathType(WeightCount)-MathType(1))/MathType(WeightCount)) // Get weight of all the older members
243  + MathType(Datum)/MathType(WeightCount); // wGet the Weight of the Current Member
244  }
