88  /// @brief Convenience function for getting the Kerning information for a given Glyph.

92  {
93  if(Kernings.size() == 0)
94  return 0;
95  for(size_t i=0;i < Kernings.size();i++)
96  {
97  if((UInt32)(Kernings[i].Character) == Left_of)
