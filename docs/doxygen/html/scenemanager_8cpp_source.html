190  SceneManager::SceneManager(XML::Node& XMLNode)
191  {
192  this->SMD = new Internal::SceneManagerData(this);
193  this->Priority = 30;
194 
195  XML::Attribute CurrAttrib;
196  // Get the name of the manager to construct.
197  String ManagerName;
198  XML::Node ManagerType = XMLNode.GetChild("ManagerType");
199  if(!ManagerType.Empty())
200  {
201  CurrAttrib = ManagerType.GetAttribute("TypeName");
202  if(!CurrAttrib.Empty())
203  ManagerName = CurrAttrib.AsString();
204  }
205  if(ManagerName.empty())
206  ManagerName = "Default";
207  this->SMD->OgreManager = Ogre::Root::getSingleton().createSceneManager(ManagerName+"SceneManager");
208 
209  // Setup the shadow configuration
210  bool TextureShadows = false;
211  XML::Node ShadowSettings = XMLNode.GetChild("ShadowSettings");
212  if(!ShadowSettings.Empty())
213  {
214  String TechniqueName;
215  CurrAttrib = ShadowSettings.GetAttribute("Technique");
216  if(!CurrAttrib.Empty())
217  {
218  TechniqueName = CurrAttrib.AsString();
219  if( "StencilModulative" == TechniqueName )
220  SetSceneShadowTechnique(SST_Stencil_Modulative);
221  else if( "StencilAdditive" == TechniqueName )
222  SetSceneShadowTechnique(SST_Stencil_Additive);
223  else if( "TextureModulative" == TechniqueName )
224  SetSceneShadowTechnique(SST_Texture_Modulative);
225  else if( "TextureAdditive" == TechniqueName )
226  SetSceneShadowTechnique(SST_Texture_Additive);
227  else if( "TextureAdditiveIntegrated" == TechniqueName )
228  SetSceneShadowTechnique(SST_Texture_Additive_Integrated);
229  else if( "TextureModulativeIntegrated" == TechniqueName )
230  SetSceneShadowTechnique(SST_Texture_Modulative_Integrated);
231  }
232 
233  if(!TechniqueName.empty())
234  {
235  if(String::npos != TechniqueName.find("Texture"))
236  TextureShadows = true;
237 
238  CurrAttrib = ShadowSettings.GetAttribute("ShadowColour");
239  if(!CurrAttrib)
240  SetShadowColour(StringTools::ConvertToColourValue(CurrAttrib.AsString()));
241 
242  CurrAttrib = ShadowSettings.GetAttribute("ShadowFarDistance");
243  if(!CurrAttrib)
244  SetShadowFarDistance(CurrAttrib.AsReal());
245  }
246  }
247 
248  // Setup texture shadow settings if any are set.
249  XML::Node TextureShadowSettings = XMLNode.GetChild("TextureShadowSettings");
250  if(!TextureShadowSettings.Empty() && TextureShadows)
251  {
252  CurrAttrib = TextureShadowSettings.GetAttribute("ShadowTextureCount");
253  if(!CurrAttrib.Empty())
254  SetShadowTextureCount(CurrAttrib.AsWhole());
255 
256  CurrAttrib = TextureShadowSettings.GetAttribute("ShadowTextureSize");
257  if(!CurrAttrib.Empty())
258  SetShadowTextureSize(static_cast<unsigned short>(CurrAttrib.AsWhole()));
259  }
260  }

281  void SceneManager::SetSceneShadowTechnique(SceneShadowTechnique Shadows)
282  {
283  Ogre::ShadowTechnique Type;
284  switch (Shadows)
285  {
286  case SST_Stencil_Modulative:
287  Type = Ogre::SHADOWTYPE_STENCIL_MODULATIVE;
288  break;
289  case SST_Stencil_Additive:
290  Type = Ogre::SHADOWTYPE_STENCIL_ADDITIVE;
291  break;
292  case SST_Texture_Modulative:
293  Type = Ogre::SHADOWTYPE_TEXTURE_MODULATIVE;
294  break;
295  case SST_Texture_Additive:
296  Type = Ogre::SHADOWTYPE_TEXTURE_ADDITIVE;
297  break;
298  case SST_Texture_Additive_Integrated:
299  Type = Ogre::SHADOWTYPE_TEXTURE_ADDITIVE_INTEGRATED;
300  break;
301  case SST_Texture_Modulative_Integrated:
302  Type = Ogre::SHADOWTYPE_TEXTURE_MODULATIVE_INTEGRATED;
303  break;
304  default:
305  Type = Ogre::SHADOWTYPE_NONE;
306  }
307  this->SMD->OgreManager->setShadowTechnique(Type);
308  }
309 
310  SceneManager::SceneShadowTechnique SceneManager::GetSceneShadowTechnique() const
311  {
312  Ogre::ShadowTechnique ShadowType = this->SMD->OgreManager->getShadowTechnique();
313  switch (ShadowType)
314  {
315  case Ogre::SHADOWTYPE_STENCIL_MODULATIVE:
316  return SST_Stencil_Modulative;
317  break;
318  case Ogre::SHADOWTYPE_STENCIL_ADDITIVE:
319  return SST_Stencil_Additive;
320  break;
321  case Ogre::SHADOWTYPE_TEXTURE_MODULATIVE:
322  return SST_Texture_Modulative;
323  break;
324  case Ogre::SHADOWTYPE_TEXTURE_ADDITIVE:
325  return SST_Texture_Additive;
326  break;
327  case Ogre::SHADOWTYPE_TEXTURE_ADDITIVE_INTEGRATED:
328  return SST_Texture_Additive_Integrated;
329  break;
330  case Ogre::SHADOWTYPE_TEXTURE_MODULATIVE_INTEGRATED:
331  return SST_Texture_Modulative_Integrated;
332  break;
333  default:
334  return SST_None;
335  }
